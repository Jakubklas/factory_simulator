7/// The variant type id is the type of the variant but without its payload.
8#[derive(Debug, Clone, Copy, PartialEq, Serialize, Deserialize)]
9pub enum VariantTypeId {
10    // Null / Empty
11    Empty,
12    // Scalar types
13    Boolean,
14    SByte,
15    Byte,
16    Int16,
17    UInt16,
18    Int32,
19    UInt32,
20    Int64,
21    UInt64,
22    Float,
23    Double,
24    String,
25    DateTime,
26    Guid,
27    StatusCode,
28    ByteString,
29    XmlElement,
30    QualifiedName,
31    LocalizedText,
32    NodeId,
33    ExpandedNodeId,
34    ExtensionObject,
35    Variant,
36    DataValue,
37    DiagnosticInfo,
38    Array,
39}
40
41impl TryFrom<&NodeId> for VariantTypeId {
42    type Error = ();
43    fn try_from(value: &NodeId) -> Result<Self, Self::Error> {
44        if value.namespace == 0 {
45            if let Identifier::Numeric(type_id) = value.identifier {
46                match type_id {
47                    type_id if type_id == DataTypeId::Boolean as u32 => Ok(VariantTypeId::Boolean),
48                    type_id if type_id == DataTypeId::Byte as u32 => Ok(VariantTypeId::Byte),
49                    type_id if type_id == DataTypeId::Int16 as u32 => Ok(VariantTypeId::Int16),
50                    type_id if type_id == DataTypeId::UInt16 as u32 => Ok(VariantTypeId::UInt16),
51                    type_id if type_id == DataTypeId::Int32 as u32 => Ok(VariantTypeId::Int32),
52                    type_id if type_id == DataTypeId::UInt32 as u32 => Ok(VariantTypeId::UInt32),
53                    type_id if type_id == DataTypeId::Int64 as u32 => Ok(VariantTypeId::Int64),
54                    type_id if type_id == DataTypeId::UInt64 as u32 => Ok(VariantTypeId::UInt64),
55                    type_id if type_id == DataTypeId::Float as u32 => Ok(VariantTypeId::Float),
56                    type_id if type_id == DataTypeId::Double as u32 => Ok(VariantTypeId::Double),
57                    type_id if type_id == DataTypeId::String as u32 => Ok(VariantTypeId::String),
58                    type_id if type_id == DataTypeId::DateTime as u32 => {
59                        Ok(VariantTypeId::DateTime)
60                    }
61                    type_id if type_id == DataTypeId::Guid as u32 => Ok(VariantTypeId::Guid),
62                    type_id if type_id == DataTypeId::ByteString as u32 => {
63                        Ok(VariantTypeId::ByteString)
64                    }
65                    type_id if type_id == DataTypeId::XmlElement as u32 => {
66                        Ok(VariantTypeId::XmlElement)
67                    }
68                    type_id if type_id == DataTypeId::NodeId as u32 => Ok(VariantTypeId::NodeId),
69                    type_id if type_id == DataTypeId::ExpandedNodeId as u32 => {
70                        Ok(VariantTypeId::ExpandedNodeId)
71                    }
72                    type_id if type_id == DataTypeId::XmlElement as u32 => {
73                        Ok(VariantTypeId::XmlElement)
74                    }
75                    type_id if type_id == DataTypeId::StatusCode as u32 => {
76                        Ok(VariantTypeId::StatusCode)
77                    }
78                    type_id if type_id == DataTypeId::QualifiedName as u32 => {
79                        Ok(VariantTypeId::QualifiedName)
80                    }
81                    type_id if type_id == DataTypeId::LocalizedText as u32 => {
82                        Ok(VariantTypeId::LocalizedText)
83                    }
84                    type_id if type_id == DataTypeId::DataValue as u32 => {
85                        Ok(VariantTypeId::DataValue)
86                    }
87                    type_id if type_id == DataTypeId::BaseDataType as u32 => {
88                        Ok(VariantTypeId::Variant)
89                    }
90                    type_id if type_id == DataTypeId::DiagnosticInfo as u32 => {
91                        Ok(VariantTypeId::DiagnosticInfo)
92                    }
93                    _ => Err(()),
94                }
95            } else {
96                Err(())
97            }
98        } else {
99            Err(())
100        }
101    }
102}

104impl VariantTypeId {
105    pub fn encoding_mask(&self) -> u8 {
106        match self {
107            // Null / Empty
108            VariantTypeId::Empty => 0u8,
109            // Scalar types
110            VariantTypeId::Boolean => EncodingMask::BOOLEAN,
111            VariantTypeId::SByte => EncodingMask::SBYTE,
112            VariantTypeId::Byte => EncodingMask::BYTE,
113            VariantTypeId::Int16 => EncodingMask::INT16,
114            VariantTypeId::UInt16 => EncodingMask::UINT16,
115            VariantTypeId::Int32 => EncodingMask::INT32,
116            VariantTypeId::UInt32 => EncodingMask::UINT32,
117            VariantTypeId::Int64 => EncodingMask::INT64,
118            VariantTypeId::UInt64 => EncodingMask::UINT64,
119            VariantTypeId::Float => EncodingMask::FLOAT,
120            VariantTypeId::Double => EncodingMask::DOUBLE,
121            VariantTypeId::String => EncodingMask::STRING,
122            VariantTypeId::DateTime => EncodingMask::DATE_TIME,
123            VariantTypeId::Guid => EncodingMask::GUID,
124            VariantTypeId::StatusCode => EncodingMask::STATUS_CODE,
125            VariantTypeId::ByteString => EncodingMask::BYTE_STRING,
126            VariantTypeId::XmlElement => EncodingMask::XML_ELEMENT,
127            VariantTypeId::QualifiedName => EncodingMask::QUALIFIED_NAME,
128            VariantTypeId::LocalizedText => EncodingMask::LOCALIZED_TEXT,
129            VariantTypeId::NodeId => EncodingMask::NODE_ID,
130            VariantTypeId::ExpandedNodeId => EncodingMask::EXPANDED_NODE_ID,
131            VariantTypeId::ExtensionObject => EncodingMask::EXTENSION_OBJECT,
132            VariantTypeId::Variant => EncodingMask::VARIANT,
133            VariantTypeId::DataValue => EncodingMask::DATA_VALUE,
134            VariantTypeId::DiagnosticInfo => EncodingMask::DIAGNOSTIC_INFO,
135            VariantTypeId::Array => panic!("Type of array is unknown"),
136        }
137    }
138
139    pub fn from_encoding_mask(encoding_mask: u8) -> Result<Self, StatusCode> {
140        match encoding_mask & !EncodingMask::ARRAY_MASK {
141            0u8 => Ok(VariantTypeId::Empty),
142            EncodingMask::BOOLEAN => Ok(VariantTypeId::Boolean),
143            EncodingMask::SBYTE => Ok(VariantTypeId::SByte),
144            EncodingMask::BYTE => Ok(VariantTypeId::Byte),
145            EncodingMask::INT16 => Ok(VariantTypeId::Int16),
146            EncodingMask::UINT16 => Ok(VariantTypeId::UInt16),
147            EncodingMask::INT32 => Ok(VariantTypeId::Int32),
148            EncodingMask::UINT32 => Ok(VariantTypeId::UInt32),
149            EncodingMask::INT64 => Ok(VariantTypeId::Int64),
150            EncodingMask::UINT64 => Ok(VariantTypeId::UInt64),
151            EncodingMask::FLOAT => Ok(VariantTypeId::Float),
152            EncodingMask::DOUBLE => Ok(VariantTypeId::Double),
153            EncodingMask::STRING => Ok(VariantTypeId::String),
154            EncodingMask::DATE_TIME => Ok(VariantTypeId::DateTime),
155            EncodingMask::GUID => Ok(VariantTypeId::Guid),
156            EncodingMask::STATUS_CODE => Ok(VariantTypeId::StatusCode),
157            EncodingMask::BYTE_STRING => Ok(VariantTypeId::ByteString),
158            EncodingMask::XML_ELEMENT => Ok(VariantTypeId::XmlElement),
159            EncodingMask::QUALIFIED_NAME => Ok(VariantTypeId::QualifiedName),
160            EncodingMask::LOCALIZED_TEXT => Ok(VariantTypeId::LocalizedText),
161            EncodingMask::NODE_ID => Ok(VariantTypeId::NodeId),
162            EncodingMask::EXPANDED_NODE_ID => Ok(VariantTypeId::ExpandedNodeId),
163            EncodingMask::EXTENSION_OBJECT => Ok(VariantTypeId::ExtensionObject),
164            EncodingMask::VARIANT => Ok(VariantTypeId::Variant),
165            EncodingMask::DATA_VALUE => Ok(VariantTypeId::DataValue),
166            EncodingMask::DIAGNOSTIC_INFO => Ok(VariantTypeId::DiagnosticInfo),
167            _ => {
168                error!("Unrecognized encoding mask");
169                Err(StatusCode::BadDecodingError)
170            }
171        }
172    }
173
174    /// Tests and returns true if the variant holds a numeric type
175    pub fn is_numeric(&self) -> bool {
176        matches!(
177            self,
178            VariantTypeId::SByte
179                | VariantTypeId::Byte
180                | VariantTypeId::Int16
181                | VariantTypeId::UInt16
182                | VariantTypeId::Int32
183                | VariantTypeId::UInt32
184                | VariantTypeId::Int64
185                | VariantTypeId::UInt64
186                | VariantTypeId::Float
187                | VariantTypeId::Double
188        )
189    }
190
191    /// Returns a data precedence rank for scalar types, OPC UA part 4 table 119. This is used
192    /// when operators are comparing values of differing types. The type with
193    /// the highest precedence dictates how values are converted in order to be compared.
194    pub fn precedence(&self) -> u8 {
195        match self {
196            VariantTypeId::Double => 1,
197            VariantTypeId::Float => 2,
198            VariantTypeId::Int64 => 3,
199            VariantTypeId::UInt64 => 4,
200            VariantTypeId::Int32 => 5,
201            VariantTypeId::UInt32 => 6,
202            VariantTypeId::StatusCode => 7,
203            VariantTypeId::Int16 => 8,
204            VariantTypeId::UInt16 => 9,
205            VariantTypeId::SByte => 10,
206            VariantTypeId::Byte => 11,
207            VariantTypeId::Boolean => 12,
208            VariantTypeId::Guid => 13,
209            VariantTypeId::String => 14,
210            VariantTypeId::ExpandedNodeId => 15,
211            VariantTypeId::NodeId => 16,
212            VariantTypeId::LocalizedText => 17,
213            VariantTypeId::QualifiedName => 18,
214            _ => 100,
215        }
216    }
217}

228    pub const INT32: u8 = DataTypeId::Int32 as u8;
229    pub const UINT32: u8 = DataTypeId::UInt32 as u8;
230    pub const INT64: u8 = DataTypeId::Int64 as u8;
231    pub const UINT64: u8 = DataTypeId::UInt64 as u8;
232    pub const FLOAT: u8 = DataTypeId::Float as u8;
233    pub const DOUBLE: u8 = DataTypeId::Double as u8;
234    pub const STRING: u8 = DataTypeId::String as u8;
235    pub const DATE_TIME: u8 = DataTypeId::DateTime as u8;
236    pub const GUID: u8 = DataTypeId::Guid as u8;
237    pub const BYTE_STRING: u8 = DataTypeId::ByteString as u8;
238    pub const XML_ELEMENT: u8 = DataTypeId::XmlElement as u8;
239    pub const NODE_ID: u8 = DataTypeId::NodeId as u8;
240    pub const EXPANDED_NODE_ID: u8 = DataTypeId::ExpandedNodeId as u8;
241    pub const STATUS_CODE: u8 = DataTypeId::StatusCode as u8;
242    pub const QUALIFIED_NAME: u8 = DataTypeId::QualifiedName as u8;
243    pub const LOCALIZED_TEXT: u8 = DataTypeId::LocalizedText as u8;
244    pub const EXTENSION_OBJECT: u8 = 22; // DataTypeId::ExtensionObject as u8;
245    pub const DATA_VALUE: u8 = DataTypeId::DataValue as u8;
246    pub const VARIANT: u8 = 24;
247    pub const DIAGNOSTIC_INFO: u8 = DataTypeId::DiagnosticInfo as u8;
248    /// Bit indicates an array with dimensions
249    pub const ARRAY_DIMENSIONS_BIT: u8 = 1 << 6;
250    /// Bit indicates an array with values
251    pub const ARRAY_VALUES_BIT: u8 = 1 << 7;
252
253    pub const ARRAY_MASK: u8 = EncodingMask::ARRAY_DIMENSIONS_BIT | EncodingMask::ARRAY_VALUES_BIT;
