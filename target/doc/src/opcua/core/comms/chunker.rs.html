23/// The Chunker is responsible for turning messages to chunks and chunks into messages.
24pub struct Chunker;
25
26impl Chunker {
27    /// Tests what kind of chunk type is used for the supported message.
28    fn message_type(message: &SupportedMessage) -> MessageChunkType {
29        match message {
30            SupportedMessage::OpenSecureChannelRequest(_)
31            | SupportedMessage::OpenSecureChannelResponse(_) => MessageChunkType::OpenSecureChannel,
32            SupportedMessage::CloseSecureChannelRequest(_)
33            | SupportedMessage::CloseSecureChannelResponse(_) => {
34                MessageChunkType::CloseSecureChannel
35            }
36            _ => MessageChunkType::Message,
37        }
38    }

40    /// Ensure all of the supplied chunks have a valid secure channel id, and sequence numbers
41    /// greater than the input sequence number and the preceding chunk
42    ///
43    /// The function returns the last sequence number in the series for success, or
44    /// `BadSequenceNumberInvalid` or `BadSecureChannelIdInvalid` for failure.
45    pub fn validate_chunks(
46        starting_sequence_number: u32,
47        secure_channel: &SecureChannel,
48        chunks: &[MessageChunk],
49    ) -> Result<u32, StatusCode> {
50        let first_sequence_number = {
51            let chunk_info = chunks[0].chunk_info(secure_channel)?;
52            chunk_info.sequence_header.sequence_number
53        };
54        if first_sequence_number < starting_sequence_number {
55            error!(
56                "First sequence number of {} is less than last value {}",
57                first_sequence_number, starting_sequence_number
58            );
59            Err(StatusCode::BadSequenceNumberInvalid)
60        } else {
61            let secure_channel_id = secure_channel.secure_channel_id();
62
63            // Validate that all chunks have incrementing sequence numbers and valid chunk types
64            let mut expected_request_id: u32 = 0;
65            for (i, chunk) in chunks.iter().enumerate() {
66                let chunk_info = chunk.chunk_info(secure_channel)?;
67
68                // Check the channel id of each chunk
69                if secure_channel_id != 0
70                    && chunk_info.message_header.secure_channel_id != secure_channel_id
71                {
72                    error!(
73                        "Secure channel id {} does not match expected id {}",
74                        chunk_info.message_header.secure_channel_id, secure_channel_id
75                    );
76                    return Err(StatusCode::BadSecureChannelIdInvalid);
77                }
78
79                // Check the sequence id - should be larger than the last one decoded
80                let sequence_number = chunk_info.sequence_header.sequence_number;
81                let expected_sequence_number = first_sequence_number + i as u32;
82                if sequence_number != expected_sequence_number {
83                    error!(
84                        "Chunk sequence number of {} is not the expected value of {}, idx {}",
85                        sequence_number, expected_sequence_number, i
86                    );
87                    return Err(StatusCode::BadSecurityChecksFailed);
88                }
89
90                // Check the request id against the first chunk's request id
91                if i == 0 {
92                    expected_request_id = chunk_info.sequence_header.request_id;
93                } else if chunk_info.sequence_header.request_id != expected_request_id {
94                    error!("Chunk sequence number of {} has a request id {} which is not the expected value of {}, idx {}", sequence_number, chunk_info.sequence_header.request_id, expected_request_id, i);
95                    return Err(StatusCode::BadSecurityChecksFailed);
96                }
97            }
98            Ok(first_sequence_number + chunks.len() as u32 - 1)
99        }
100    }

102    /// Encodes a message using the supplied sequence number and secure channel info and emits the corresponding chunks
103    ///
104    /// max_chunk_count refers to the maximum byte length that a chunk should not exceed or 0 for no limit
105    /// max_message_size refers to the maximum byte length of a message or 0 for no limit
106    ///
107    pub fn encode(
108        sequence_number: u32,
109        request_id: u32,
110        max_message_size: usize,
111        max_chunk_size: usize,
112        secure_channel: &SecureChannel,
113        supported_message: &SupportedMessage,
114    ) -> std::result::Result<Vec<MessageChunk>, StatusCode> {
115        let security_policy = secure_channel.security_policy();
116        if security_policy == SecurityPolicy::Unknown {
117            panic!("Security policy cannot be unknown");
118        }
119
120        // Client / server stacks should validate the length of a message before sending it and
121        // here makes as good a place as any to do that.
122        let mut message_size = supported_message.byte_len();
123        if max_message_size > 0 && message_size > max_message_size {
124            error!(
125                "Max message size is {} and message {} exceeds that",
126                max_message_size, message_size
127            );
128            // Client stack should report a BadRequestTooLarge, server BadResponseTooLarge
129            Err(if secure_channel.is_client_role() {
130                StatusCode::BadRequestTooLarge
131            } else {
132                StatusCode::BadResponseTooLarge
133            })
134        } else {
135            let node_id = supported_message.node_id();
136            message_size += node_id.byte_len();
137
138            let message_type = Chunker::message_type(supported_message);
139            let mut stream = Cursor::new(vec![0u8; message_size]);
140
141            trace!("Encoding node id {:?}", node_id);
142            let _ = node_id.encode(&mut stream);
143            let _ = supported_message.encode(&mut stream)?;
144            let data = stream.into_inner();
145
146            let result = if max_chunk_size > 0 {
147                let max_body_per_chunk = MessageChunk::body_size_from_message_size(
148                    message_type,
149                    secure_channel,
150                    max_chunk_size,
151                )
152                .map_err(|_| {
153                    error!(
154                        "body_size_from_message_size error for max_chunk_size = {}",
155                        max_chunk_size
156                    );
157                    StatusCode::BadTcpInternalError
158                })?;
159
160                // Multiple chunks means breaking the data up into sections. Fortunately
161                // Rust has a nice function to do just that.
162                let data_chunks = data.chunks(max_body_per_chunk);
163                let data_chunks_len = data_chunks.len();
164                let mut chunks = Vec::with_capacity(data_chunks_len);
165                for (i, data_chunk) in data_chunks.enumerate() {
166                    let is_final = if i == data_chunks_len - 1 {
167                        MessageIsFinalType::Final
168                    } else {
169                        MessageIsFinalType::Intermediate
170                    };
171                    let chunk = MessageChunk::new(
172                        sequence_number + i as u32,
173                        request_id,
174                        message_type,
175                        is_final,
176                        secure_channel,
177                        data_chunk,
178                    )?;
179                    chunks.push(chunk);
180                }
181                chunks
182            } else {
183                let chunk = MessageChunk::new(
184                    sequence_number,
185                    request_id,
186                    message_type,
187                    MessageIsFinalType::Final,
188                    secure_channel,
189                    &data,
190                )?;
191                vec![chunk]
192            };
193            Ok(result)
194        }
195    }

197    /// Decodes a series of chunks to create a message. The message must be of a `SupportedMessage`
198    /// type otherwise an error will occur.
199    pub fn decode(
200        chunks: &[MessageChunk],
201        secure_channel: &SecureChannel,
202        expected_node_id: Option<NodeId>,
203    ) -> std::result::Result<SupportedMessage, StatusCode> {
204        // Calculate the size of data held in all chunks
205        let mut data_size: usize = 0;
206        for (i, chunk) in chunks.iter().enumerate() {
207            let chunk_info = chunk.chunk_info(secure_channel)?;
208            // The last most chunk is expected to be final, the rest intermediate
209            let expected_is_final = if i == chunks.len() - 1 {
210                MessageIsFinalType::Final
211            } else {
212                MessageIsFinalType::Intermediate
213            };
214            if chunk_info.message_header.is_final != expected_is_final {
215                return Err(StatusCode::BadDecodingError);
216            }
217            // Calculate how much space data is in the chunk
218            let body_start = chunk_info.body_offset;
219            let body_end = body_start + chunk_info.body_length;
220            data_size += chunk.data[body_start..body_end].len();
221        }
222
223        // Read the data into a contiguous buffer. The assumption is the data is decrypted / verified by now
224        // TODO this buffer should be externalized so it is not allocated each time
225        let mut data = Vec::with_capacity(data_size);
226        for chunk in chunks.iter() {
227            let chunk_info = chunk.chunk_info(secure_channel)?;
228
229            let body_start = chunk_info.body_offset;
230            let body_end = body_start + chunk_info.body_length;
231            let body_data = &chunk.data[body_start..body_end];
232            data.extend_from_slice(body_data);
233        }
234
235        // Make a stream around the data
236        let mut data = Cursor::new(data);
237
238        // The extension object prefix is just the node id. A point the spec rather unhelpfully doesn't
239        // elaborate on. Probably because people enjoy debugging why the stream pos is out by 1 byte
240        // for hours.
241
242        let decoding_options = secure_channel.decoding_options();
243
244        // Read node id from stream
245        let node_id = NodeId::decode(&mut data, &decoding_options)?;
246        let object_id = Self::object_id_from_node_id(node_id, expected_node_id)?;
247
248        // Now decode the payload using the node id.
249        match SupportedMessage::decode_by_object_id(&mut data, object_id, &decoding_options) {
250            Ok(decoded_message) => {
251                if let SupportedMessage::Invalid(_) = decoded_message {
252                    debug!("Message {:?} is unsupported", object_id);
253                    Err(StatusCode::BadServiceUnsupported)
254                } else {
255                    // debug!("Returning decoded msg {:?}", decoded_message);
256                    Ok(decoded_message)
257                }
258            }
259            Err(err) => {
260                debug!("Cannot decode message {:?}, err = {:?}", object_id, err);
261                Err(StatusCode::BadServiceUnsupported)
262            }
263        }
264    }
265
266    fn object_id_from_node_id(
267        node_id: NodeId,
268        expected_node_id: Option<NodeId>,
269    ) -> Result<ObjectId, StatusCode> {
270        let valid_node_id = if node_id.namespace != 0 || !node_id.is_numeric() {
271            // Must be ns 0 and numeric
272            error!("Expecting chunk to contain a OPC UA request or response");
273            false
274        } else if let Some(expected_node_id) = expected_node_id {
275            let matches_expected = expected_node_id == node_id;
276            if !matches_expected {
277                error!(
278                    "Chunk node id {:?} does not match expected {:?}",
279                    node_id, expected_node_id
280                );
281            }
282            matches_expected
283        } else {
284            true
285        };
286        if !valid_node_id {
287            error!(
288                "The node id read from the stream was not accepted in this context {:?}",
289                node_id
290            );
291            Err(StatusCode::BadUnexpectedError)
292        } else {
293            node_id
294                .as_object_id()
295                .map_err(|_| {
296                    error!("The node {:?} was not an object id", node_id);
297                    StatusCode::BadUnexpectedError
298                })
299                .map(|object_id| {
300                    trace!("Decoded node id / object id of {:?}", object_id);
301                    object_id
302                })
303        }
304    }
