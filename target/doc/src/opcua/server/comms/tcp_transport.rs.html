5//! The TCP transport module handles receiving and sending of binary data in chunks, handshake,
6//! session creation and dispatching of messages via message handler.
7//!
8//! Internally it uses tokio but the facade is mostly synchronous with the exception of publish
9//! responses. i.e. the client is expected to call and wait for a response to their request.
10//! Publish requests are sent based on the number of subscriptions and the responses / handling are
11//! left to asynchronous event handlers.

50/// Messages that may be sent to the writer.
51#[derive(Debug)]
52enum Message {
53    // Message for writer to quit right now.
54    Quit,
55    // A supported message with a request id
56    Message(u32, SupportedMessage),
57}
58
59#[derive(Clone)]
60pub struct MessageSender {
61    sender: UnboundedSender<Message>,
62}
63
64impl MessageSender {
65    pub fn send_quit(&self) {
66        let _ = self.sender.send(Message::Quit);
67    }
68
69    pub fn send_message(&self, request_id: u32, message: SupportedMessage) {
70        let _ = self.sender.send(Message::Message(request_id, message));
71    }
72}

96/// This is the thing that handles input and output for the open connection associated with the
97/// session.
98pub struct TcpTransport {
99    /// Server state, address space etc.
100    server_state: Arc<RwLock<ServerState>>,
101    /// Transport id (for debugging)
102    transport_id: NodeId,
103    /// Secure channel state
104    secure_channel: Arc<RwLock<SecureChannel>>,
105    /// Address space
106    address_space: Arc<RwLock<AddressSpace>>,
107    /// The current transport state
108    transport_state: TransportState,
109    /// Client address
110    client_address: Option<SocketAddr>,
111    /// Secure channel handler
112    secure_channel_service: SecureChannelService,
113    /// Message handler
114    message_handler: MessageHandler,
115    /// Client protocol version set during HELLO
116    client_protocol_version: u32,
117    /// Last decoded sequence number
118    last_received_sequence_number: u32,
119    /// A message may consist of one or more chunks which are stored here until complete.
120    pending_chunks: Vec<MessageChunk>,
121    /// Sessions associated with this connection. Normally there would be one, but potentially there could be more
122    session_manager: Arc<RwLock<SessionManager>>,
123}
124
125impl Transport for TcpTransport {
126    fn state(&self) -> TransportState {
127        self.transport_state
128    }
129
130    // Terminates the connection and the session
131    fn finish(&mut self, status_code: StatusCode) {
132        if !self.is_finished() {
133            debug!(
134                "Transport is being placed in finished state, code {}",
135                status_code
136            );
137            self.transport_state = TransportState::Finished(status_code);
138            // Clear sessions
139            let mut session_manager = trace_write_lock!(self.session_manager);
140            session_manager.clear(self.address_space.clone());
141        } else {
142            trace!("Transport is being placed in finished state when it is already finished, ignoring code {}", status_code);
143        }
144    }
145
146    fn client_address(&self) -> Option<SocketAddr> {
147        self.client_address
148    }
149
150    fn session_manager(&self) -> Arc<RwLock<SessionManager>> {
151        self.session_manager.clone()
152    }
153}

157        certificate_store: Arc<RwLock<CertificateStore>>,
158        server_state: Arc<RwLock<ServerState>>,
159        address_space: Arc<RwLock<AddressSpace>>,
160        session_manager: Arc<RwLock<SessionManager>>,

163            let server_state = trace_read_lock!(server_state);
164            let config = trace_read_lock!(server_state.config);
165            config.decoding_options()
166        };
167        let secure_channel = Arc::new(RwLock::new(SecureChannel::new(
168            certificate_store.clone(),
169            Role::Server,
170            decoding_options,
171        )));
172
173        let message_handler = MessageHandler::new(
174            secure_channel.clone(),
175            certificate_store,
176            server_state.clone(),
177            session_manager.clone(),
178            address_space.clone(),
179        );
180
181        let secure_channel_service = SecureChannelService::new();
182        let transport_id = NodeId::next_numeric(0);
183
184        TcpTransport {
185            server_state,
186            transport_id,
187            address_space,
188            transport_state: TransportState::New,
189            client_address: None,
190            message_handler,
191            secure_channel,
192            secure_channel_service,
193            client_protocol_version: 0,
194            last_received_sequence_number: 0,
195            pending_chunks: Vec::with_capacity(2),
196            session_manager,
197        }

200    /// This is the entry point for the session. This function is asynchronous - it spawns tokio
201    /// tasks to handle the session execution loop so this function will returns immediately.
202    pub fn run(connection: Arc<RwLock<TcpTransport>>, socket: TcpStream, looping_interval_ms: f64) {
203        info!(
204            "Socket info:\n  Linger - {},\n  TTL - {}",
205            if let Ok(v) = socket.linger() {
206                match v {
207                    Some(d) => format!("{}ms", d.as_millis()),
208                    None => "No linger".to_string(),
209                }
210            } else {
211                "No Linger (err)".to_string()
212            },
213            if let Ok(v) = socket.ttl() {
214                format!("{}", v)
215            } else {
216                "No TTL".to_string()
217            }
218        );
219
220        // Store the address of the client
221        let (send_buffer_size, receive_buffer_size) = {
222            let mut connection = trace_write_lock!(connection);
223            connection.client_address = Some(socket.peer_addr().unwrap());
224            connection.transport_state = TransportState::WaitingHello;
225            let server_state = trace_read_lock!(connection.server_state);
226            (
227                server_state.send_buffer_size,
228                server_state.receive_buffer_size,
229            )
230        };
231
232        // Spawn the tasks we need to run
233        tokio::spawn(Self::spawn_session_handler_task(
234            connection,
235            socket,
236            looping_interval_ms,
237            send_buffer_size,
238            receive_buffer_size,
239        ));
240    }

242    async fn write_bytes_task(mut write_state: WriteState) -> WriteState {
243        let bytes_to_write = {
244            let mut send_buffer = trace_lock!(write_state.send_buffer);
245            send_buffer.bytes_to_write()
246        };
247        let result = write_state.writer.write_all(&bytes_to_write).await;
248        if let Err(err) = result {
249            error!("Write IO error {:?}", err);
250            let mut transport = trace_write_lock!(write_state.transport);
251            transport.finish(StatusCode::BadCommunicationError);
252        }
253        write_state
254    }
255
256    async fn spawn_session_handler_task(
257        transport: Arc<RwLock<TcpTransport>>,
258        socket: TcpStream,
259        looping_interval_ms: f64,
260        send_buffer_size: usize,
261        receive_buffer_size: usize,
262    ) {
263        // The reader task will send responses, the writer task will receive responses
264        let (tx, rx) = unbounded_channel();
265        let send_buffer = Arc::new(Mutex::new(MessageWriter::new(send_buffer_size, 0, 0)));
266
267        let (reader, writer) = socket.into_split();
268        let (hello_timeout, secure_channel) = {
269            let transport = trace_read_lock!(transport);
270            let server_state = trace_read_lock!(transport.server_state);
271            let server_config = trace_read_lock!(server_state.config);
272            info!(
273                "Session transport {} started at {}",
274                transport.transport_id,
275                Utc::now()
276            );
277            (
278                server_config.tcp_config.hello_timeout,
279                transport.secure_channel.clone(),
280            )
281        };
282
283        let read_state = ReadState {
284            reader,
285            hello_timeout,
286            transport: transport.clone(),
287            sender: tx.clone(),
288        };
289
290        // Spawn all the tasks that monitor the session - the subscriptions, finished state,
291        // reading and writing.
292        let final_status = tokio::select! {
293            _ = Self::spawn_subscriptions_task(transport.clone(), tx.clone(), looping_interval_ms) => {
294                log::trace!("Closing connection because the subscription task failed");
295                Ok(())
296            }
297            status = Self::spawn_writing_loop_task(writer, rx, secure_channel, transport.clone(), send_buffer) => {
298                log::trace!("Closing connection after the write task ended");
299                status
300            }
301            status = Self::spawn_reading_loop_task(read_state, send_buffer_size, receive_buffer_size) => {
302                log::trace!("Closing connection after the read task ended");
303                status
304            }
305        }.err().unwrap_or(StatusCode::Good);
306
307        log::info!("Closing connection with status {}", final_status);
308        // Both the read and write halves of the tcp stream are dropped at this point,
309        // and the connection is closed
310        let mut transport = trace_write_lock!(transport);
311        transport.finish(final_status);
312    }

314    /// Spawns the writing loop task. The writing loop takes messages to send off of a queue
315    /// and sends them to the stream.
316    async fn spawn_writing_loop_task(
317        writer: OwnedWriteHalf,
318        mut receiver: UnboundedReceiver<Message>,
319        secure_channel: Arc<RwLock<SecureChannel>>,
320        transport: Arc<RwLock<TcpTransport>>,
321        send_buffer: Arc<Mutex<MessageWriter>>,
322    ) -> Result<(), StatusCode> {
323        let mut write_state = WriteState {
324            transport: transport.clone(),
325            writer,
326            send_buffer,
327            secure_channel,
328        };
329
330        // The writing task waits for messages that are to be sent
331        while let Some(message) = receiver.recv().await {
332            trace!("Writing loop received message: {:?}", message);
333            let (request_id, response) = match message {
334                Message::Quit => {
335                    debug!("Server writer received a quit so it will quit");
336                    return Ok(());
337                }
338                Message::Message(request_id, response) => {
339                    if let SupportedMessage::Invalid(_) = response {
340                        error!("Writer terminating - received an invalid message");
341                        return Err(StatusCode::BadCommunicationError);
342                    }
343                    (request_id, response)
344                }
345            };
346
347            {
348                let secure_channel = trace_read_lock!(write_state.secure_channel);
349                let mut send_buffer = trace_lock!(write_state.send_buffer);
350                match response {
351                    SupportedMessage::AcknowledgeMessage(ack) => {
352                        send_buffer.write_ack(&ack)?;
353                    }
354                    msg => {
355                        send_buffer.write(request_id, msg, &secure_channel)?;
356                    }
357                }
358            }
359            write_state = Self::write_bytes_task(write_state).await;
360        }
361        Ok(())
362    }

364    async fn wait_for_hello(
365        reader: &mut FramedRead<OwnedReadHalf, TcpCodec>,
366        hello_timeout: u32,
367    ) -> Result<HelloMessage, StatusCode> {
368        let duration = Duration::from_secs(u64::from(hello_timeout));
369        match timeout(duration, reader.next()).await {
370            // We process a timeout(stream_element(tcp_message))
371            Err(_timeout) => {
372                warn!("Session has been waiting for a hello for more than the timeout period and will now close");
373                Err(StatusCode::BadTimeout)
374            }
375            Ok(Some(Ok(tcp_codec::Message::Hello(hello)))) => Ok(hello),
376            Ok(Some(Ok(bad_msg))) => {
377                log::error!("Expected a hello message, got {:?} instead", bad_msg);
378                Err(StatusCode::BadCommunicationError)
379            }
380            Ok(Some(Err(communication_err))) => {
381                error!(
382                    "Communication error while waiting for Hello message: {}",
383                    communication_err
384                );
385                Err(StatusCode::BadCommunicationError)
386            }
387            Ok(None) => Err(StatusCode::BadConnectionClosed),
388        }
389    }

391    /// Spawns the reading loop where a reader task continuously reads messages, chunks from the
392    /// input and process them. The reading task will terminate upon error.
393    async fn spawn_reading_loop_task(
394        read_state: ReadState,
395        send_buffer_size: usize,
396        receive_buffer_size: usize,
397    ) -> Result<(), StatusCode> {
398        let (transport, mut sender) = { (read_state.transport.clone(), read_state.sender.clone()) };
399
400        let decoding_options = {
401            let transport = trace_read_lock!(transport);
402            let secure_channel = trace_read_lock!(transport.secure_channel);
403            secure_channel.decoding_options()
404        };
405
406        // The reader reads frames from the codec, which are messages
407        let mut framed_read =
408            FramedRead::new(read_state.reader, TcpCodec::new(decoding_options.clone()));
409
410        let hello = Self::wait_for_hello(&mut framed_read, read_state.hello_timeout).await?;
411        trace_write_lock!(transport).process_hello(
412            hello,
413            &mut sender,
414            &decoding_options,
415            send_buffer_size,
416            receive_buffer_size,
417        )?;
418
419        while let Some(next_msg) = framed_read.next().await {
420            match next_msg {
421                Ok(tcp_codec::Message::Chunk(chunk)) => {
422                    log::trace!("Received message chunk: {:?}", chunk);
423                    let mut transport = trace_write_lock!(transport);
424                    transport.process_chunk(chunk, &mut sender)?
425                }
426                Ok(unexpected) => {
427                    log::error!("Received unexpected message: {:?}", unexpected);
428                    return Err(StatusCode::BadCommunicationError);
429                }
430                Err(err) => {
431                    error!("Server reader error {:?}", err);
432                    return Err(StatusCode::BadCommunicationError);
433                }
434            }
435        }
436        Ok(())
437    }

439    /// Start the subscription timer to service subscriptions
440    async fn spawn_subscriptions_task(
441        transport: Arc<RwLock<TcpTransport>>,
442        sender: UnboundedSender<Message>,
443        looping_interval_ms: f64,
444    ) -> Result<(), StatusCode> {
445        // Subscription events are passed sent from the monitor task to the receiver
446        debug!("Starting subscription timer loop");
447
448        // Create the monitoring timer - this monitors for publish requests and ticks the subscriptions
449        let interval_duration = Duration::from_millis(looping_interval_ms as u64);
450
451        // Creates a repeating interval future that checks subscriptions.
452        let mut timer = interval_at(Instant::now(), interval_duration);
453
454        loop {
455            timer.tick().await;
456
457            let transport = trace_read_lock!(transport);
458            let session_manager = trace_read_lock!(transport.session_manager);
459
460            for (_node_id, session) in session_manager.sessions.iter() {
461                let mut session = trace_write_lock!(session);
462                let address_space = trace_read_lock!(transport.address_space);
463                let now = Utc::now();
464
465                // Request queue might contain stale publish requests
466                session.expire_stale_publish_requests(&now);
467
468                // Process subscriptions
469                session.tick_subscriptions(&now, &address_space, TickReason::TickTimerFired)?;
470
471                // Check if there are publish responses to send for transmission
472                if let Some(publish_responses) =
473                    session.subscriptions_mut().take_publish_responses()
474                {
475                    for publish_response in publish_responses {
476                        trace!(
477                            "<-- Sending a Publish Response{}, {:?}",
478                            publish_response.request_id,
479                            &publish_response.response
480                        );
481                        // Messages will be sent by the writing task
482                        sender
483                            .send(Message::Message(
484                                publish_response.request_id,
485                                publish_response.response,
486                            ))
487                            .map_err(|e| {
488                                error!("Unable to send publish response to writer task: {}", e);
489                                StatusCode::BadUnexpectedError
490                            })?;
491                    }
492                }
493            }
494        }
495    }

503    fn process_hello(
504        &mut self,
505        hello: HelloMessage,
506        sender: &mut UnboundedSender<Message>,
507        decoding_options: &DecodingOptions,
508        send_buffer_size: usize,
509        receive_buffer_size: usize,
510    ) -> std::result::Result<(), StatusCode> {
511        let server_protocol_version = 0;
512        let endpoints = {
513            let server_state = trace_read_lock!(self.server_state);
514            server_state.endpoints(&hello.endpoint_url, &None)
515        }
516        .unwrap();
517
518        trace!("Server received HELLO {:?}", hello);
519        if !hello.is_endpoint_url_valid(&endpoints) {
520            error!("HELLO endpoint url is invalid");
521            return Err(StatusCode::BadTcpEndpointUrlInvalid);
522        }
523        if !hello.is_valid_buffer_sizes() {
524            error!("HELLO buffer sizes are invalid");
525            return Err(StatusCode::BadCommunicationError);
526        }
527
528        // Validate protocol version
529        if hello.protocol_version > server_protocol_version {
530            return Err(StatusCode::BadProtocolVersionUnsupported);
531        }
532
533        let client_protocol_version = hello.protocol_version;
534
535        // Send acknowledge
536        let mut acknowledge = AcknowledgeMessage {
537            message_header: MessageHeader::new(MessageType::Acknowledge),
538            protocol_version: server_protocol_version,
539            receive_buffer_size: receive_buffer_size as u32,
540            send_buffer_size: send_buffer_size as u32,
541            max_message_size: decoding_options.max_message_size as u32,
542            max_chunk_count: decoding_options.max_chunk_count as u32,
543        };
544        acknowledge.message_header.message_size = acknowledge.byte_len() as u32;
545        let acknowledge: SupportedMessage = acknowledge.into();
546
547        // New state
548        self.transport_state = TransportState::ProcessMessages;
549        self.client_protocol_version = client_protocol_version;
550
551        debug!("Sending ACK");
552        let _ = sender.send(Message::Message(0, acknowledge));
553        Ok(())
554    }

556    fn turn_received_chunks_into_message(
557        &mut self,
558        chunks: &[MessageChunk],
559    ) -> std::result::Result<SupportedMessage, StatusCode> {
560        // Validate that all chunks have incrementing sequence numbers and valid chunk types
561        let secure_channel = trace_read_lock!(self.secure_channel);
562        self.last_received_sequence_number = Chunker::validate_chunks(
563            self.last_received_sequence_number + 1,
564            &secure_channel,
565            chunks,
566        )?;
567        // Now decode
568        Chunker::decode(chunks, &secure_channel, None)
569    }
570
571    fn process_chunk(
572        &mut self,
573        chunk: MessageChunk,
574        sender: &mut UnboundedSender<Message>,
575    ) -> std::result::Result<(), StatusCode> {
576        let decoding_options = {
577            let secure_channel = trace_read_lock!(self.secure_channel);
578            secure_channel.decoding_options()
579        };
580
581        let message_header = chunk.message_header(&decoding_options)?;
582
583        if message_header.is_final == MessageIsFinalType::FinalError {
584            info!("Discarding chunks as after receiving one marked as final error");
585            self.pending_chunks.clear();
586            Ok(())
587        } else {
588            // Decrypt / verify chunk if necessary
589            let chunk = {
590                let mut secure_channel = trace_write_lock!(self.secure_channel);
591                secure_channel.verify_and_remove_security(&chunk.data)?
592            };
593
594            // TODO check how many chunks are pending, produce error and drop connection if it exceeds
595            //  supported chunk limit
596
597            // Put the chunk on the list
598            self.pending_chunks.push(chunk);
599
600            // The final chunk will trigger turning all pending chunks into a request
601            if message_header.is_final == MessageIsFinalType::Final {
602                self.process_final_chunk(&message_header, sender)
603            } else {
604                Ok(())
605            }
606        }
607    }
608
609    fn process_final_chunk(
610        &mut self,
611        message_header: &MessageChunkHeader,
612        sender: &mut UnboundedSender<Message>,
613    ) -> Result<(), StatusCode> {
614        // Drain pending chunks and turn them into a message
615        let chunks: Vec<MessageChunk> = self.pending_chunks.drain(..).collect();
616        let chunk_info = {
617            let secure_channel = trace_read_lock!(self.secure_channel);
618            chunks[0].chunk_info(&secure_channel)?
619        };
620
621        // Handle the request, and then send the response back to the caller
622        let request = self.turn_received_chunks_into_message(&chunks)?;
623        let request_id = chunk_info.sequence_header.request_id;
624
625        let sender = MessageSender {
626            sender: sender.clone(),
627        };
628
629        match message_header.message_type {
630            MessageChunkType::OpenSecureChannel => self.process_open_secure_channel(
631                request_id,
632                &request,
633                &chunk_info.security_header,
634                &sender,
635            ),
636            MessageChunkType::CloseSecureChannel => {
637                self.process_close_secure_channel(request_id, &request, &sender)
638            }
639            MessageChunkType::Message => self.process_message(request_id, &request, &sender),
640        }
641    }
642
643    fn process_open_secure_channel(
644        &mut self,
645        request_id: u32,
646        request: &SupportedMessage,
647        security_header: &SecurityHeader,
648        sender: &MessageSender,
649    ) -> Result<(), StatusCode> {
650        let mut secure_channel = trace_write_lock!(self.secure_channel);
651        let response = self.secure_channel_service.open_secure_channel(
652            &mut secure_channel,
653            security_header,
654            self.client_protocol_version,
655            request,
656        )?;
657        let _ = sender.send_message(request_id, response);
658        Ok(())
659    }
660
661    fn process_close_secure_channel(
662        &mut self,
663        request_id: u32,
664        request: &SupportedMessage,
665        sender: &MessageSender,
666    ) -> Result<(), StatusCode> {
667        let response = self.secure_channel_service.close_secure_channel(request)?;
668        let _ = sender.send_message(request_id, response);
669        Ok(())
670    }
671
672    fn process_message(
673        &mut self,
674        request_id: u32,
675        request: &SupportedMessage,
676        sender: &MessageSender,
677    ) -> Result<(), StatusCode> {
678        let _ = self
679            .message_handler
680            .handle_message(request_id, request, sender)?;
681        Ok(())
682    }
