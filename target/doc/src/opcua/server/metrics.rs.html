5//! Provides debug metric of server state that can be used by anything that wants
6//! to see what is happening in the server. State is updated by the server as sessions are added, removed,
7//! and when subscriptions / monitored items are added, removed.
8
9use crate::runtime_components;
10use crate::types::DateTime;
11
12use crate::server::{
13    comms::transport::{Transport, TransportState},
14    config,
15    diagnostics::ServerDiagnostics,
16    server,
17    state::ServerState,
18    subscriptions::subscriptions,
19};
20
21#[derive(Serialize)]
22pub struct ServerMetrics {
23    pub server: Server,
24    pub diagnostics: ServerDiagnostics,
25    pub config: Option<config::ServerConfig>,
26    pub connections: Vec<Connection>,
27    pub runtime_components: Vec<String>,
28}
29
30#[derive(Serialize)]
31pub struct Server {
32    pub start_time: String,
33    pub uptime_ms: i64,
34}
35
36#[derive(Serialize)]
37pub struct Connection {
38    pub sessions: Vec<Session>,
39    // creation time
40    // state
41    pub client_address: String,
42    pub transport_state: String,
43}
44
45#[derive(Serialize)]
46pub struct Session {
47    pub id: String,
48    pub session_activated: bool,
49    pub session_terminated: bool,
50    pub session_terminated_at: String,
51    pub subscriptions: subscriptions::Metrics,
52}
53
54impl Default for ServerMetrics {
55    fn default() -> Self {
56        // Sample metrics
57        Self {
58            server: Server {
59                start_time: String::new(),
60                uptime_ms: 0,
61            },
62            diagnostics: ServerDiagnostics::default(),
63            config: None,
64            connections: Vec::new(),
65            runtime_components: Vec::new(),
66        }
67    }
68}
69
70impl ServerMetrics {
71    pub fn new() -> ServerMetrics {
72        Self::default()
73    }
74
75    pub fn set_server_info(&mut self, server: &server::Server) {
76        let server_state = server.server_state();
77        let config = {
78            let server_state = trace_read_lock!(server_state);
79            server_state.config.clone()
80        };
81        let mut config = {
82            let config = trace_read_lock!(config);
83            config.clone()
84        };
85        // For security, blank out user tokens
86        config.user_tokens.clear();
87        config.user_tokens.insert(
88            String::new(),
89            config::ServerUserToken {
90                user: String::from("User identity tokens have been removed"),
91                pass: None,
92                x509: None,
93                thumbprint: None,
94            },
95        );
96        self.config = Some(config);
97    }
98
99    // Update the server state metrics (uptime etc.)
100    pub fn update_from_server_state(&mut self, server_state: &ServerState) {
101        let start_time = &server_state.start_time;
102        let now = DateTime::now();
103
104        self.server.start_time = start_time.as_chrono().to_rfc3339();
105
106        // Take a snapshot of the diagnostics
107        {
108            let diagnostics = trace_read_lock!(server_state.diagnostics);
109            self.diagnostics = diagnostics.clone();
110        }
111
112        let elapsed = now
113            .as_chrono()
114            .signed_duration_since(start_time.as_chrono());
115        self.server.uptime_ms = elapsed.num_milliseconds();
116    }
117
118    // Update the connection metrics which includes susbcriptions and monitored items
119    pub fn update_from_connections(&mut self, connections: server::Connections) {
120        self.runtime_components = runtime_components!();
121        self.connections = connections
122            .iter()
123            .map(|c| {
124                // Carefully extract info while minimizing chance of deadlock
125                let (client_address, transport_state, session_manager) = {
126                    let connection = trace_read_lock!(c);
127                    let client_address =
128                        if let Some(ref client_address) = connection.client_address() {
129                            format!("{:?}", client_address)
130                        } else {
131                            String::new()
132                        };
133                    let transport_state = match connection.state() {
134                        TransportState::New => "New".to_string(),
135                        TransportState::WaitingHello => "WaitingHello".to_string(),
136                        TransportState::ProcessMessages => "ProcessMessages".to_string(),
137                        TransportState::Finished(status_code) => {
138                            format!("Finished({})", status_code)
139                        }
140                    };
141                    (
142                        client_address,
143                        transport_state,
144                        connection.session_manager(),
145                    )
146                };
147                let session_manager = trace_read_lock!(session_manager);
148                let sessions = session_manager
149                    .sessions
150                    .iter()
151                    .map(|(_, session)| {
152                        let session = trace_read_lock!(session);
153                        let id = session.session_id().to_string();
154                        let session_activated = session.is_activated();
155                        let session_terminated = session.is_terminated();
156                        let session_terminated_at = if session.is_terminated() {
157                            session.terminated_at().to_rfc3339()
158                        } else {
159                            String::new()
160                        };
161                        let subscriptions = session.subscriptions().metrics();
162                        Session {
163                            id,
164                            session_activated,
165                            session_terminated,
166                            session_terminated_at,
167                            subscriptions,
168                        }
169                    })
170                    .collect();
171
172                // session.subscriptions.iterate ...
173                Connection {
174                    client_address,
175                    transport_state,
176                    sessions,
177                }
178            })
179            .collect();
180    }
181}
