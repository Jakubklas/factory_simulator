16/// Given a browse path consisting of browse names, walk nodes from the root until we find a single node (or not).
17/// This function is a simplified use case for event filters and such like where a browse path
18/// is defined as an array and doesn't need to be parsed out of a relative path. All nodes in the
19/// path must be objects or variables.
20pub(crate) fn find_node_from_browse_path<'a>(
21    address_space: &'a AddressSpace,
22    parent_node_id: &NodeId,
23    browse_path: &[QualifiedName],
24) -> Result<&'a NodeType, StatusCode> {
25    if browse_path.is_empty() {
26        Err(StatusCode::BadNotFound)
27    } else {
28        // Each instance declaration in the path shall be an object or variable node. The final node in the
29        // path may be an object node; however, object nodes are only available for Events which are
30        // visible in the server's address space
31        let mut parent_node_id = parent_node_id.clone();
32        for browse_name in browse_path {
33            if let Some(child_nodes) = address_space.find_hierarchical_references(&parent_node_id) {
34                let found_node_id = child_nodes.iter().find(|node_id| {
35                    if let Some(node) = address_space.find_node(node_id) {
36                        if node.as_node().browse_name() == *browse_name {
37                            // Check that the node is an Object or Variable
38                            matches!(node, NodeType::Object(_) | NodeType::Variable(_))
39                        } else {
40                            false
41                        }
42                    } else {
43                        false
44                    }
45                });
46                if let Some(found_node_id) = found_node_id {
47                    parent_node_id = found_node_id.clone();
48                } else {
49                    //debug!(
50                    //    "Cannot find node under {} with browse_path of {:?}/1",
51                    //    parent_node_id, browse_path
52                    //);
53                    return Err(StatusCode::BadNotFound);
54                }
55            } else {
56                //debug!(
57                //    "Cannot find node under {} with browse_path of {:?}/2",
58                //    parent_node_id, browse_path
59                //);
60                return Err(StatusCode::BadNotFound);
61            }
62        }
63        Ok(address_space.find_node(&parent_node_id).unwrap())
64    }
65}

67/// Given a path as a string, find all the nodes that match against it. Note this function
68/// uses a default path resolver based on common browse names. If you need something else use
69/// `find_nodes_relative_path()` after you have created a relative path.
70pub fn find_nodes_relative_path_simple(
71    address_space: &AddressSpace,
72    node_id: &NodeId,
73    relative_path: &str,
74) -> Result<Vec<NodeId>, StatusCode> {
75    let relative_path =
76        RelativePath::from_str(relative_path, &RelativePathElement::default_node_resolver)
77            .map_err(|_| StatusCode::BadUnexpectedError)?;
78    find_nodes_relative_path(address_space, node_id, &relative_path)
79}

81/// Given a `RelativePath`, find all the nodes that match against it.
82pub fn find_nodes_relative_path(
83    address_space: &AddressSpace,
84    node_id: &NodeId,
85    relative_path: &RelativePath,
86) -> Result<Vec<NodeId>, StatusCode> {
87    match address_space.find_node(node_id) {
88        None => {
89            trace!("find_nodes_relative_path cannot find node {}", node_id);
90            Err(StatusCode::BadNodeIdUnknown)
91        }
92        Some(_) => {
93            let elements = relative_path.elements.as_ref().unwrap();
94            if elements.is_empty() {
95                warn!("find_nodes_relative_path elements are empty");
96                Err(StatusCode::BadNothingToDo)
97            } else {
98                let mut matching_nodes = vec![node_id.clone()];
99                let mut next_matching_nodes = Vec::with_capacity(100);
100
101                // Traverse the relative path elements. Each time around, we will find the matching
102                // elements at that level using the next element
103                for element in elements.iter() {
104                    if element.target_name.is_null() {
105                        warn!("find_nodes_relative_path browse name is invalid (null)");
106                        return Err(StatusCode::BadBrowseNameInvalid);
107                    }
108
109                    next_matching_nodes.clear();
110
111                    matching_nodes.drain(..).for_each(|node_id| {
112                        trace!("Following relative path on node {}", node_id);
113                        // Iterate current set of nodes and put the results into next
114                        if let Some(mut result) =
115                            follow_relative_path(address_space, &node_id, element)
116                        {
117                            trace!("  Found matching nodes {:#?}", result);
118                            next_matching_nodes.append(&mut result);
119                        } else {
120                            trace!("  Found no matching nodes");
121                        }
122                    });
123                    if next_matching_nodes.is_empty() {
124                        break;
125                    } else {
126                        matching_nodes.append(&mut next_matching_nodes);
127                    }
128                }
129
130                if matching_nodes.is_empty() {
131                    warn!("find_nodes_relative_path bad no match");
132                    Err(StatusCode::BadNoMatch)
133                } else {
134                    Ok(matching_nodes)
135                }
136            }
137        }
138    }
139}

141fn follow_relative_path(
142    address_space: &AddressSpace,
143    node_id: &NodeId,
144    relative_path: &RelativePathElement,
145) -> Option<Vec<NodeId>> {
146    let reference_filter = {
147        if let Ok(reference_type_id) = relative_path.reference_type_id.as_reference_type_id() {
148            Some((reference_type_id, relative_path.include_subtypes))
149        } else {
150            None
151        }
152    };
153    let references = if relative_path.is_inverse {
154        address_space.find_inverse_references(node_id, reference_filter)
155    } else {
156        address_space.find_references(node_id, reference_filter)
157    };
158    if let Some(references) = references {
159        let compare_target_name = !relative_path.target_name.is_null();
160        let mut result = Vec::with_capacity(references.len());
161        for reference in &references {
162            if let Some(node) = address_space.find_node(&reference.target_node) {
163                let node = node.as_node();
164                if !compare_target_name || node.browse_name() == relative_path.target_name {
165                    result.push(reference.target_node.clone());
166                }
167            }
168        }
169        // Vector may contain duplicates, so reduce those to a unique set
170        let result = result.into_iter().collect::<HashSet<NodeId>>();
171        // Now the result as a vec
172        Some(result.into_iter().collect())
173    } else {
174        None
175    }
176}
