122    /// Deregisters a session object
123    pub(crate) fn deregister_session(&self, session: &Session, address_space: &mut AddressSpace) {
124        address_space.delete(session.session_id(), true);
125    }
126
127    /// Called on every request
128    pub(crate) fn request(&mut self) {
129        self.total_request_count += 1;
130    }
131
132    /// Called on an authorized request
133    pub(crate) fn unauthorized_request(&mut self) {
134        self.unauthorized_request_count += 1;
135        self.total_request_count += 1;
136    }
137
138    /// Fetches a snapshot of the current service counter value
139    pub(crate) fn service_counter(
140        &mut self,
141        diagnostic_key: &'static str,
142    ) -> ServiceCounterDataType {
143        if let Some(counter) = self.service_counters.get_mut(diagnostic_key) {
144            counter.clone()
145        } else {
146            ServiceCounterDataType::default()
147        }
148    }
149
150    /// Increments the service counter for a successful service call
151    pub(crate) fn service_success(&mut self, diagnostic_key: &'static str) {
152        if let Some(counter) = self.service_counters.get_mut(diagnostic_key) {
153            counter.success();
154        } else {
155            let mut counter = ServiceCounterDataType::default();
156            counter.success();
157            self.service_counters.insert(diagnostic_key, counter);
158        }
159    }
160
161    /// Increments the service counter for a failed service call
162    pub(crate) fn service_error(&mut self, diagnostic_key: &'static str) {
163        if let Some(counter) = self.service_counters.get_mut(diagnostic_key) {
164            counter.error();
165        } else {
166            let mut counter = ServiceCounterDataType::default();
167            counter.error();
168            self.service_counters.insert(diagnostic_key, counter);
169        }
170    }
