15/// The monitored item service. Allows client to create, modify and delete monitored items on a subscription.
16pub(crate) struct MonitoredItemService;

21    }
22}

26        MonitoredItemService {}

29    /// Implementation of CreateMonitoredItems service. See OPC Unified Architecture, Part 4 5.12.2
30    pub fn create_monitored_items(
31        &self,
32        server_state: Arc<RwLock<ServerState>>,
33        session: Arc<RwLock<Session>>,
34        address_space: Arc<RwLock<AddressSpace>>,
35        request: &CreateMonitoredItemsRequest,
36    ) -> SupportedMessage {
37        if is_empty_option_vec!(request.items_to_create) {
38            self.service_fault(&request.request_header, StatusCode::BadNothingToDo)
39        } else {
40            let server_state = trace_read_lock!(server_state);
41            let mut session = trace_write_lock!(session);
42            let address_space = trace_read_lock!(address_space);
43
44            let items_to_create = request.items_to_create.as_ref().unwrap();
45            // Find subscription and add items to it
46            if let Some(subscription) = session.subscriptions_mut().get_mut(request.subscription_id)
47            {
48                let now = chrono::Utc::now();
49                let results = Some(subscription.create_monitored_items(
50                    &server_state,
51                    &address_space,
52                    &now,
53                    request.timestamps_to_return,
54                    items_to_create,
55                ));
56                let response = CreateMonitoredItemsResponse {
57                    response_header: ResponseHeader::new_good(&request.request_header),
58                    results,
59                    diagnostic_infos: None,
60                };
61                response.into()
62            } else {
63                // No matching subscription
64                self.service_fault(
65                    &request.request_header,
66                    StatusCode::BadSubscriptionIdInvalid,
67                )
68            }
69        }
70    }

112    /// Implementation of SetMonitoringMode service. See OPC Unified Architecture, Part 4 5.12.4
113    pub fn set_monitoring_mode(
114        &self,
115        session: Arc<RwLock<Session>>,
116        request: &SetMonitoringModeRequest,
117    ) -> SupportedMessage {
118        if is_empty_option_vec!(request.monitored_item_ids) {
119            self.service_fault(&request.request_header, StatusCode::BadNothingToDo)
120        } else {
121            let mut session = trace_write_lock!(session);
122            let monitored_item_ids = request.monitored_item_ids.as_ref().unwrap();
123            let subscription_id = request.subscription_id;
124            if let Some(subscription) = session.subscriptions_mut().get_mut(subscription_id) {
125                let monitoring_mode = request.monitoring_mode;
126                let results = monitored_item_ids
127                    .iter()
128                    .map(|i| subscription.set_monitoring_mode(*i, monitoring_mode))
129                    .collect();
130                SetMonitoringModeResponse {
131                    response_header: ResponseHeader::new_good(&request.request_header),
132                    results: Some(results),
133                    diagnostic_infos: None,
134                }
135                .into()
136            } else {
137                self.service_fault(
138                    &request.request_header,
139                    StatusCode::BadSubscriptionIdInvalid,
140                )
141            }
142        }
143    }

145    /// Implementation of SetTriggering service. See OPC Unified Architecture, Part 4 5.12.5
146    pub fn set_triggering(
147        &self,
148        session: Arc<RwLock<Session>>,
149        request: &SetTriggeringRequest,
150    ) -> SupportedMessage {
151        if is_empty_option_vec!(request.links_to_add)
152            && is_empty_option_vec!(request.links_to_remove)
153        {
154            self.service_fault(&request.request_header, StatusCode::BadNothingToDo)
155        } else {
156            let mut session = trace_write_lock!(session);
157            let links_to_add = match request.links_to_add {
158                Some(ref links_to_add) => &links_to_add[..],
159                None => &[],
160            };
161            let links_to_remove = match request.links_to_remove {
162                Some(ref links_to_remove) => &links_to_remove[..],
163                None => &[],
164            };
165
166            // Set the triggering on the subscription.
167            let subscription_id = request.subscription_id;
168            if let Some(subscription) = session.subscriptions_mut().get_mut(subscription_id) {
169                match subscription.set_triggering(
170                    request.triggering_item_id,
171                    links_to_add,
172                    links_to_remove,
173                ) {
174                    Ok((add_results, remove_results)) => {
175                        let response = SetTriggeringResponse {
176                            response_header: ResponseHeader::new_good(&request.request_header),
177                            add_results: if request.links_to_add.is_some() {
178                                Some(add_results)
179                            } else {
180                                None
181                            },
182                            add_diagnostic_infos: None,
183                            remove_results: if request.links_to_remove.is_some() {
184                                Some(remove_results)
185                            } else {
186                                None
187                            },
188                            remove_diagnostic_infos: None,
189                        };
190                        response.into()
191                    }
192                    Err(err) => self.service_fault(&request.request_header, err),
193                }
194            } else {
195                self.service_fault(
196                    &request.request_header,
197                    StatusCode::BadSubscriptionIdInvalid,
198                )
199            }
200        }
201    }

203    /// Implementation of DeleteMonitoredItems service. See OPC Unified Architecture, Part 4 5.12.6
204    pub fn delete_monitored_items(
205        &self,
206        session: Arc<RwLock<Session>>,
207        request: &DeleteMonitoredItemsRequest,
208    ) -> SupportedMessage {
209        if is_empty_option_vec!(request.monitored_item_ids) {
210            self.service_fault(&request.request_header, StatusCode::BadNothingToDo)
211        } else {
212            let mut session = trace_write_lock!(session);
213            let monitored_item_ids = request.monitored_item_ids.as_ref().unwrap();
214            // Find subscription and delete items from it
215            let subscription_id = request.subscription_id;
216            if let Some(subscription) = session.subscriptions_mut().get_mut(subscription_id) {
217                let results = Some(subscription.delete_monitored_items(monitored_item_ids));
218                let diagnostic_infos = None;
219                let response = DeleteMonitoredItemsResponse {
220                    response_header: ResponseHeader::new_good(&request.request_header),
221                    results,
222                    diagnostic_infos,
223                };
224                response.into()
225            } else {
226                // No matching subscription
227                self.service_fault(
228                    &request.request_header,
229                    StatusCode::BadSubscriptionIdInvalid,
230                )
231            }
232        }
233    }
