5//! This module implements the audit event types.
6//!
7//! Note: Due to Rust's lack of inheritance, these types use aggregation and helper macros to expose
8//! builder functions on each type in the hierarchy. They're not optimal at all (impls call base impls call base impls in some cases),
9//! but they should suffice for the purpose they'll be used for.
10
11use std::sync::Arc;
12
13use crate::sync::*;
14use crate::types::*;
15
16use crate::server::{address_space::address_space::AddressSpace, events::event::Event};
17
18pub trait AuditEvent: Event {
19    fn parent_node() -> NodeId {
20        // TODO Where do audit nodes get put in the address_space?
21        NodeId::null()
22    }
23
24    /// Returns the kind of event type that this audit event represents. Abstract events should
25    /// panic.
26    fn event_type_id() -> NodeId;
27
28    fn log_message(&self) -> String;
29}
30
31#[macro_use]
32pub mod event;
33#[macro_use]
34pub mod security_event;
35#[macro_use]
36pub mod session_events;
37#[macro_use]
38pub mod certificate_events;
39pub mod cancel_event;
40pub mod node_management_event;

42/// The audit log will be responsible for adding audit events to the address space, and potentially logging them
43/// to file. All audit events should be raised through `AuditLog` to support any future logging capability.
44pub(crate) struct AuditLog {
45    address_space: Arc<RwLock<AddressSpace>>,
46}
47
48impl AuditLog {
49    pub fn new(address_space: Arc<RwLock<AddressSpace>>) -> AuditLog {
50        AuditLog { address_space }
51    }
52
53    pub fn raise_and_log<T>(&self, mut event: T) -> Result<NodeId, ()>
54    where
55        T: AuditEvent + Event,
56    {
57        let mut address_space = trace_write_lock!(self.address_space);
58        let result = event.raise(&mut address_space).map_err(|_| ());
59        if result.is_err() {
60            error!("Cannot raise an audit event, check audit event entry below to see if there are reasons for this");
61        }
62        // At this point audit events just go out as log events but smarter logging implementations can always hive these
63        // events off to a separate file. Look at demo-server for an example of this.
64        info!("Audit Event: {}", event.log_message());
65        result
66    }
67}
