23/// This validates the event filter as best it can to make sure it doesn't contain nonsense.
24pub fn validate(
25    event_filter: &EventFilter,
26    address_space: &AddressSpace,
27) -> Result<EventFilterResult, StatusCode> {
28    let select_clause_results = event_filter.select_clauses.as_ref().map(|select_clauses| {
29        select_clauses
30            .iter()
31            .map(|clause| validate_select_clause(clause, address_space))
32            .collect()
33    });
34    let where_clause_result = validate_where_clause(&event_filter.where_clause, address_space)?;
35    Ok(EventFilterResult {
36        select_clause_results,
37        select_clause_diagnostic_infos: None,
38        where_clause_result,
39    })
40}

42/// Evaluate the event filter and see if it triggers.
43pub fn evaluate(
44    object_id: &NodeId,
45    event_filter: &EventFilter,
46    address_space: &AddressSpace,
47    happened_since: &DateTimeUtc,
48    client_handle: u32,
49) -> Option<Vec<EventFieldList>> {
50    if let Some(events) = events_for_object(object_id, address_space, happened_since) {
51        let event_fields = events
52            .iter()
53            .filter(|event_id| {
54                if let Ok(result) =
55                    evaluate_where_clause(event_id, &event_filter.where_clause, address_space)
56                {
57                    result == Variant::Boolean(true)
58                } else {
59                    false
60                }
61            })
62            .map(|event_id| {
63                // Produce an event notification list from the select clauses.
64                let event_fields = event_filter.select_clauses.as_ref().map(|select_clauses| {
65                    select_clauses
66                        .iter()
67                        .map(|v| operator::value_of_simple_attribute(event_id, v, address_space))
68                        .collect()
69                });
70                EventFieldList {
71                    client_handle,
72                    event_fields,
73                }
74            })
75            .collect::<Vec<EventFieldList>>();
76        if event_fields.is_empty() {
77            None
78        } else {
79            Some(event_fields)
80        }
81    } else {
82        None
83    }
84}

86/// Evaluates a where clause which is a tree of conditionals
87pub(crate) fn evaluate_where_clause(
88    object_id: &NodeId,
89    where_clause: &ContentFilter,
90    address_space: &AddressSpace,
91) -> Result<Variant, StatusCode> {
92    // Clause is meant to have been validated before now so this code is not as stringent and makes some expectations.
93    if let Some(ref elements) = where_clause.elements {
94        if !elements.is_empty() {
95            use std::collections::HashSet;
96            let mut used_elements = HashSet::new();
97            used_elements.insert(0);
98            let result = operator::evaluate(
99                object_id,
100                &elements[0],
101                &mut used_elements,
102                elements,
103                address_space,
104            )?;
105            Ok(result)
106        } else {
107            Ok(true.into())
108        }
109    } else {
110        Ok(true.into())
111    }
112}

114fn validate_select_clause(
115    clause: &SimpleAttributeOperand,
116    address_space: &AddressSpace,
117) -> StatusCode {
118    // The SimpleAttributeOperand structure is used in the selectClauses to select the value to return
119    // if an Event meets the criteria specified by the whereClause. A null value is returned in the corresponding
120    // event field in the publish response if the selected field is not part of the event or an
121    // error was returned in the selectClauseResults of the EventFilterResult.
122
123    if !clause.index_range.is_empty() {
124        // TODO support index ranges
125        error!("Select clause specifies an index range and will be rejected");
126        StatusCode::BadIndexRangeInvalid
127    } else if let Some(ref browse_path) = clause.browse_path {
128        // Validate that the browse paths seem okay relative to the object type definition in the clause
129        if let Ok(node) =
130            find_node_from_browse_path(address_space, &clause.type_definition_id, browse_path)
131        {
132            // Validate the attribute id. Per spec:
133            //
134            //   The SimpleAttributeOperand allows the client to specify any attribute; however the server
135            //   is only required to support the value attribute for variable nodes and the NodeId attribute
136            //   for object nodes. That said, profiles defined in Part 7 may make support for
137            //   additional attributes mandatory.
138            //
139            // So code will implement the bare minimum for now.
140            let valid_attribute_id = match node {
141                NodeType::Object(_) => {
142                    // Only the node id
143                    clause.attribute_id == AttributeId::NodeId as u32
144                }
145                NodeType::Variable(_) => {
146                    // Only the value
147                    clause.attribute_id == AttributeId::Value as u32
148                }
149                _ => {
150                    // find_node_from_browse_path shouldn't have returned anything except an object
151                    // or variable node.
152                    panic!()
153                }
154            };
155            if !valid_attribute_id {
156                StatusCode::BadAttributeIdInvalid
157            } else {
158                StatusCode::Good
159            }
160        } else {
161            error!("Invalid select clause node not found {:?}", clause);
162            StatusCode::BadNodeIdUnknown
163        }
164    } else {
165        error!("Invalid select clause with no browse path supplied");
166        StatusCode::BadNodeIdUnknown
167    }
168}

170fn validate_where_clause(
171    where_clause: &ContentFilter,
172    address_space: &AddressSpace,
173) -> Result<ContentFilterResult, StatusCode> {
174    // The ContentFilter structure defines a collection of elements that define filtering criteria.
175    // Each element in the collection describes an operator and an array of operands to be used by
176    // the operator. The operators that can be used in a ContentFilter are described in Table 119.
177    // The filter is evaluated by evaluating the first entry in the element array starting with the
178    // first operand in the operand array. The operands of an element may contain References to
179    // sub-elements resulting in the evaluation continuing to the referenced elements in the element
180    // array. The evaluation shall not introduce loops. For example evaluation starting from element
181    // “A” shall never be able to return to element “A”. However there may be more than one path
182    // leading to another element “B”. If an element cannot be traced back to the starting element
183    // it is ignored. Extra operands for any operator shall result in an error. Annex B provides
184    // examples using the ContentFilter structure.
185
186    if let Some(ref elements) = where_clause.elements {
187        let element_results = elements.iter().map(|e| {
188            let (status_code, operand_status_codes) = if e.filter_operands.is_none() {
189                // All operators need at least one operand
190                (StatusCode::BadFilterOperandCountMismatch, None)
191            } else {
192                let filter_operands = e.filter_operands.as_ref().unwrap();
193
194                // The right number of operators? The spec implies it is okay to pass
195                // more operands than the required #, but less is an error.
196                let operand_count_mismatch = match e.filter_operator {
197                    FilterOperator::Equals => filter_operands.len() < 2,
198                    FilterOperator::IsNull => filter_operands.len() < 1,
199                    FilterOperator::GreaterThan => filter_operands.len() < 2,
200                    FilterOperator::LessThan => filter_operands.len() < 2,
201                    FilterOperator::GreaterThanOrEqual => filter_operands.len() < 2,
202                    FilterOperator::LessThanOrEqual => filter_operands.len() < 2,
203                    FilterOperator::Like => filter_operands.len() < 2,
204                    FilterOperator::Not => filter_operands.len() < 1,
205                    FilterOperator::Between => filter_operands.len() < 3,
206                    FilterOperator::InList => filter_operands.len() < 2, // 2..n
207                    FilterOperator::And => filter_operands.len() < 2,
208                    FilterOperator::Or => filter_operands.len() < 2,
209                    FilterOperator::Cast => filter_operands.len() < 2,
210                    FilterOperator::BitwiseAnd => filter_operands.len() < 2,
211                    FilterOperator::BitwiseOr => filter_operands.len() < 2,
212                    _ => true,
213                };
214
215                // Check if the operands look okay
216                let operand_status_codes = filter_operands.iter().map(|e| {
217                    // Look to see if any operand cannot be parsed
218                    match <Operand>::try_from(e) {
219                        Ok(operand) => {
220                            match operand {
221                                Operand::AttributeOperand(_) => {
222                                    // AttributeOperand may not be used in an EventFilter where clause
223                                    error!("AttributeOperand is not permitted in EventFilter where clause");
224                                    StatusCode::BadFilterOperandInvalid
225                                }
226                                Operand::ElementOperand(ref o) => {
227                                    // Check that operands have to have an index <= number of elements
228                                    if o.index as usize >= elements.len() {
229                                        error!("Invalid element operand is out of range");
230                                        StatusCode::BadFilterOperandInvalid
231                                    } else {
232                                        StatusCode::Good
233                                    }
234                                    // TODO operand should not refer to itself either directly or through circular
235                                    //  references
236                                }
237                                Operand::SimpleAttributeOperand(ref o) => {
238                                    // The structure requires the node id of an event type supported
239                                    //  by the server and a path to an InstanceDeclaration. An InstanceDeclaration
240                                    //  is a Node which can be found by following forward hierarchical references from the fully
241                                    //  inherited EventType where the Node is also the source of a HasModellingRuleReference. EventTypes
242                                    //  InstanceDeclarations and Modelling rules are described completely in Part 3.
243
244                                    // In some case the same BrowsePath will apply to multiple EventTypes. If
245                                    // the Client specifies the BaseEventType in the SimpleAttributeOperand
246                                    // then the Server shall evaluate the BrowsePath without considering the Type.
247
248                                    // Each InstanceDeclaration in the path shall be Object or Variable Node.
249
250                                    // Check the element exists in the address space
251                                    if let Some(ref browse_path) = o.browse_path {
252                                        if let Ok(_node) = find_node_from_browse_path(address_space, &o.type_definition_id, browse_path) {
253                                            StatusCode::Good
254                                        } else {
255                                            StatusCode::BadFilterOperandInvalid
256                                        }
257                                    } else {
258                                        StatusCode::BadFilterOperandInvalid
259                                    }
260                                }
261                                _ => StatusCode::Good
262                            }
263                        }
264                        Err(err) => {
265                            error!("Operand cannot be read from extension object, err = {}", err);
266                            StatusCode::BadFilterOperandInvalid
267                        }
268                    }
269                }).collect::<Vec<StatusCode>>();
270
271                // Check if any operands were invalid
272                let operator_invalid = operand_status_codes.iter().any(|e| !e.is_good());
273
274                // Check what error status to return
275                let status_code = if operand_count_mismatch {
276                    error!("Where clause has invalid filter operand count");
277                    StatusCode::BadFilterOperandCountMismatch
278                } else if operator_invalid {
279                    error!("Where clause has invalid filter operator");
280                    StatusCode::BadFilterOperatorInvalid
281                } else {
282                    StatusCode::Good
283                };
284
285                (status_code, Some(operand_status_codes))
286            };
287            ContentFilterElementResult {
288                status_code,
289                operand_status_codes,
290                operand_diagnostic_infos: None,
291            }
292        }).collect::<Vec<ContentFilterElementResult>>();
293
294        Ok(ContentFilterResult {
295            element_results: Some(element_results),
296            element_diagnostic_infos: None,
297        })
298    } else {
299        Ok(ContentFilterResult {
300            element_results: None,
301            element_diagnostic_infos: None,
302        })
303    }
304}

306#[test]
307fn validate_where_clause_test() {
308    use crate::types::service_types::ContentFilterElement;
309
310    let address_space = AddressSpace::new();
311
312    {
313        let where_clause = ContentFilter { elements: None };
314        // check for at least one filter operand
315        let result = validate_where_clause(&where_clause, &address_space);
316        assert_eq!(
317            result.unwrap(),
318            ContentFilterResult {
319                element_results: None,
320                element_diagnostic_infos: None,
321            }
322        );
323    }
324
325    // Make a where clause where every single operator is included but each has the wrong number of operands.
326    // We should expect them all to be in error
327    {
328        let where_clause = ContentFilter {
329            elements: Some(vec![
330                ContentFilterElement::from((FilterOperator::Equals, vec![Operand::literal(10)])),
331                ContentFilterElement::from((FilterOperator::IsNull, vec![])),
332                ContentFilterElement::from((
333                    FilterOperator::GreaterThan,
334                    vec![Operand::literal(10)],
335                )),
336                ContentFilterElement::from((FilterOperator::LessThan, vec![Operand::literal(10)])),
337                ContentFilterElement::from((
338                    FilterOperator::GreaterThanOrEqual,
339                    vec![Operand::literal(10)],
340                )),
341                ContentFilterElement::from((
342                    FilterOperator::LessThanOrEqual,
343                    vec![Operand::literal(10)],
344                )),
345                ContentFilterElement::from((FilterOperator::Like, vec![Operand::literal(10)])),
346                ContentFilterElement::from((FilterOperator::Not, vec![])),
347                ContentFilterElement::from((
348                    FilterOperator::Between,
349                    vec![Operand::literal(10), Operand::literal(20)],
350                )),
351                ContentFilterElement::from((FilterOperator::InList, vec![Operand::literal(10)])),
352                ContentFilterElement::from((FilterOperator::And, vec![Operand::literal(10)])),
353                ContentFilterElement::from((FilterOperator::Or, vec![Operand::literal(10)])),
354                ContentFilterElement::from((FilterOperator::Cast, vec![Operand::literal(10)])),
355                ContentFilterElement::from((
356                    FilterOperator::BitwiseAnd,
357                    vec![Operand::literal(10)],
358                )),
359                ContentFilterElement::from((FilterOperator::BitwiseOr, vec![Operand::literal(10)])),
360                ContentFilterElement::from((FilterOperator::Like, vec![Operand::literal(10)])),
361            ]),
362        };
363        // Check for less than required number of operands
364        let result = validate_where_clause(&where_clause, &address_space).unwrap();
365        result
366            .element_results
367            .unwrap()
368            .iter()
369            .for_each(|e| assert_eq!(e.status_code, StatusCode::BadFilterOperandCountMismatch));
370    }
371
372    // check for filter operator invalid, by giving it a bogus extension object for an element
373    {
374        use crate::types::{service_types::ContentFilterElement, ExtensionObject};
375        let bad_operator = ExtensionObject::null();
376        let where_clause = ContentFilter {
377            elements: Some(vec![ContentFilterElement {
378                filter_operator: FilterOperator::IsNull,
379                filter_operands: Some(vec![bad_operator]),
380            }]),
381        };
382        let result = validate_where_clause(&where_clause, &address_space).unwrap();
383        let element_results = result.element_results.unwrap();
384        assert_eq!(element_results.len(), 1);
385        assert_eq!(
386            element_results[0].status_code,
387            StatusCode::BadFilterOperatorInvalid
388        );
389    }
