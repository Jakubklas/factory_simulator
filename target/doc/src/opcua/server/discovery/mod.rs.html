13/// Returns the directory where the UA-LDS service stores its certs on Windows
14fn windows_lds_pki_dir() -> String {
15    /// Default derived from https://github.com/OPCFoundation/UA-LDS/blob/master/win32/platform.c
16    const WINDOWS_LDS_PKI_DIR: &str = r#"C:\ProgramData\OPC Foundation\UA\pki"#;
17
18    if cfg!(windows) {
19        // On Windows the logic can check the environment variable like UA-LDS does
20        if let Ok(mut pki_dir) = std::env::var("ALLUSERSPROFILE") {
21            pki_dir.push_str(r#"\OPC Foundation\UA\pki"#);
22            pki_dir
23        } else {
24            WINDOWS_LDS_PKI_DIR.to_string()
25        }
26    } else {
27        WINDOWS_LDS_PKI_DIR.to_string()
28    }
29}
30
31/// Returns the directory where the UA-LDS service stores its certs on Linux
32fn linux_lds_pki_dir() -> String {
33    /// Derived from https://github.com/OPCFoundation/UA-LDS/blob/master/linux/platform.c
34    const LINUX_LDS_PKI_DIR: &str = "/opt/opcfoundation/ualds/pki";
35    LINUX_LDS_PKI_DIR.to_string()
36}

38/// Registers the specified endpoints with the specified discovery server
39pub fn register_with_discovery_server(discovery_server_url: &str, server_state: &ServerState) {
40    debug!(
41        "register_with_discovery_server, for {}",
42        discovery_server_url
43    );
44    let server_config = trace_read_lock!(server_state.config);
45
46    // Create a client, ensuring to retry only once
47    let client = ClientBuilder::new()
48        .application_name("DiscoveryClient")
49        .application_uri("urn:DiscoveryClient")
50        .pki_dir(server_config.pki_dir.clone())
51        .session_retry_limit(1)
52        .client();
53
54    if let Some(mut client) = client {
55        // This follows the local discovery process described in part 12 of the spec, calling
56        // find_servers on it first.
57
58        // Connect to the server and call find_servers to ensure it is a discovery server
59        match client.find_servers(discovery_server_url) {
60            Ok(servers) => {
61                debug!("Servers on the discovery endpoint - {:?}", servers);
62                // Register the server
63                let registered_server = server_state.registered_server();
64                match client.register_server(discovery_server_url, registered_server) {
65                    Ok(_) => {}
66                    Err(err) => {
67                        error!(
68                            r#"Cannot register server with discovery server \"{}\".
69The errors immediately preceding this message may be caused by this issue.
70Check if the error "{}" indicates the reason why that the registration could not happen.
71
72Check that your server can connect to the discovery server and that your server's cert is trusted by
73the discovery server and vice versa. The discovery server's PKI directory is (Windows)
74{} or (Linux) {}."#,
75                            discovery_server_url,
76                            err,
77                            windows_lds_pki_dir(),
78                            linux_lds_pki_dir()
79                        );
80                    }
81                }
82            }
83            Err(err) => {
84                error!(
85                    "Cannot find servers on discovery url {}, error = {:?}",
86                    discovery_server_url, err
87                );
88            }
89        }
90    } else {
91        error!("Cannot create a discovery server client config");
92    }
93
94    debug!("register_with_discovery_server, finished");
95}
