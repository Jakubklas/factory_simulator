5//! Functions related to encrypting / decrypting passwords in a UserNameIdentityToken.

7//! The code here determines how or if to encrypt the password depending on the security policy

22/// Create a filled in UserNameIdentityToken by using the supplied channel security policy, user token policy, nonce, cert, user name and password.
23pub fn make_user_name_identity_token(
24    channel_security_policy: SecurityPolicy,
25    user_token_policy: &UserTokenPolicy,
26    nonce: &[u8],
27    cert: &Option<X509>,
28    user: &str,
29    pass: &str,
30) -> Result<UserNameIdentityToken, StatusCode> {
31    // This is a condensed version of Table 187 Opc Part 4 that details the EncryptionAlgorithm
32    // selection.
33    //
34    // This is mostly along the lines of: The UserTokenPolicy.SecurityPolicy takes precedence over the
35    // SecureChannel.SecurityPolicy, except there's a distinction between the cases when
36    // UserTokenPolicy.SecurityPolicy is null/empty and explicitly set to SecurityPolicy::None. In
37    // the first case, the SecureChannel.SecurityPolicy is to be used, and in the latter case the
38    // policy is explicitly set to None.
39    //
40    let security_policy = if user_token_policy.security_policy_uri.is_empty() {
41        // If no SecurityPolicy is explicitly set for UserIdentityToken, use the one defined in
42        // SecureChannel.
43        channel_security_policy
44    } else {
45        let security_policy =
46            SecurityPolicy::from_str(user_token_policy.security_policy_uri.as_ref()).unwrap();
47        if security_policy == SecurityPolicy::Unknown {
48            SecurityPolicy::None
49        } else {
50            security_policy
51        }
52    };
53
54    // Now it should be a matter of using the policy (or lack thereof) to encrypt the password
55    // using the secure channel's cert and nonce.
56    let (password, encryption_algorithm) = match security_policy {
57        SecurityPolicy::None => {
58            // Plain text
59            if channel_security_policy == SecurityPolicy::None {
60                warn!("A user identity's password is being sent over the network in plain text. This could be a serious security issue");
61            }
62            (ByteString::from(pass.as_bytes()), UAString::null())
63        }
64        SecurityPolicy::Unknown => {
65            // This should only happen if channel_security_policy were Unknown when it shouldn't be
66            panic!("Don't know how to make the token for this server");
67        }
68        security_policy => {
69            // Create a password which is encrypted using the secure channel info and the user token policy for the endpoint
70            let password = legacy_password_encrypt(
71                pass,
72                nonce,
73                cert.as_ref().unwrap(),
74                security_policy.asymmetric_encryption_padding(),
75            )?;
76            let encryption_algorithm =
77                UAString::from(security_policy.asymmetric_encryption_algorithm());
78            (password, encryption_algorithm)
79        }
80    };
81
82    Ok(UserNameIdentityToken {
83        policy_id: user_token_policy.policy_id.clone(),
84        user_name: UAString::from(user),
85        password,
86        encryption_algorithm,
87    })
88}

90/// Decrypt the password inside of a user identity token.
91pub fn decrypt_user_identity_token_password(
92    user_identity_token: &UserNameIdentityToken,
93    server_nonce: &[u8],
94    server_key: &PrivateKey,
95) -> Result<String, StatusCode> {
96    if user_identity_token.encryption_algorithm.is_empty() {
97        // Assumed to be UTF-8 plain text
98        user_identity_token.plaintext_password()
99    } else {
100        // Determine the padding from the algorithm.
101        let encryption_algorithm = user_identity_token.encryption_algorithm.as_ref();
102        let padding = match encryption_algorithm {
103            super::algorithms::ENC_RSA_15 => RsaPadding::Pkcs1,
104            super::algorithms::ENC_RSA_OAEP => RsaPadding::OaepSha1,
105            super::algorithms::ENC_RSA_OAEP_SHA256 => RsaPadding::OaepSha256,
106            _ => {
107                error!("decrypt_user_identity_token_password has rejected unsupported user identity encryption algorithm \"{}\"", encryption_algorithm);
108                return Err(StatusCode::BadIdentityTokenInvalid);
109            }
110        };
111        legacy_password_decrypt(
112            &user_identity_token.password,
113            server_nonce,
114            server_key,
115            padding,
116        )
117    }
118}

120/// Encrypt a client side user's password using the server nonce and cert. This is described in table 176
121/// OPC UA part 4. This function is prefixed "legacy" because 1.04 describes another way of encrypting passwords.
122pub fn legacy_password_encrypt(
123    password: &str,
124    server_nonce: &[u8],
125    server_cert: &X509,
126    padding: RsaPadding,
127) -> Result<ByteString, StatusCode> {
128    // Message format is size, password, nonce
129    let plaintext_size = 4 + password.len() + server_nonce.len();
130    let mut src = Cursor::new(vec![0u8; plaintext_size]);
131
132    // Write the length of the data to be encrypted excluding the length itself)
133    write_u32(&mut src, (plaintext_size - 4) as u32)?;
134    src.write(password.as_bytes())
135        .map_err(|_| StatusCode::BadEncodingError)?;
136    src.write(server_nonce)
137        .map_err(|_| StatusCode::BadEncodingError)?;
138
139    // Encrypt the data with the public key from the server's certificate
140    let public_key = server_cert.public_key()?;
141
142    let cipher_size = public_key.calculate_cipher_text_size(plaintext_size, padding);
143    let mut dst = vec![0u8; cipher_size];
144    let actual_size = public_key
145        .public_encrypt(&src.into_inner(), &mut dst, padding)
146        .map_err(|_| StatusCode::BadEncodingError)?;
147
148    assert_eq!(actual_size, cipher_size);
149
150    Ok(ByteString::from(dst))
151}

153/// Decrypt the client's password using the server's nonce and private key. This function is prefixed
154/// "legacy" because 1.04 describes another way of encrypting passwords.
155pub fn legacy_password_decrypt(
156    secret: &ByteString,
157    server_nonce: &[u8],
158    server_key: &PrivateKey,
159    padding: RsaPadding,
160) -> Result<String, StatusCode> {
161    if secret.is_null() {
162        Err(StatusCode::BadDecodingError)
163    } else {
164        // Decrypt the message
165        let src = secret.value.as_ref().unwrap();
166        let mut dst = vec![0u8; src.len()];
167        let actual_size = server_key
168            .private_decrypt(src, &mut dst, padding)
169            .map_err(|_| StatusCode::BadEncodingError)?;
170
171        let mut dst = Cursor::new(dst);
172        let plaintext_size = read_u32(&mut dst)? as usize;
173        if plaintext_size + 4 != actual_size {
174            Err(StatusCode::BadDecodingError)
175        } else {
176            let dst = dst.into_inner();
177            let nonce_len = server_nonce.len();
178            let nonce_begin = actual_size - nonce_len;
179            let nonce = &dst[nonce_begin..(nonce_begin + nonce_len)];
180            if nonce != server_nonce {
181                Err(StatusCode::BadDecodingError)
182            } else {
183                let password = &dst[4..nonce_begin];
184                let password = String::from_utf8(password.to_vec())
185                    .map_err(|_| StatusCode::BadEncodingError)?;
186                Ok(password)
187            }
188        }
189    }
190}

192/// Verify that the X509 identity token supplied to a server contains a valid signature.
193pub fn verify_x509_identity_token(
194    token: &X509IdentityToken,
195    user_token_signature: &SignatureData,
196    security_policy: SecurityPolicy,
197    server_cert: &X509,
198    server_nonce: &[u8],
199) -> Result<(), StatusCode> {
200    // Since it is not obvious at all from the spec what the user token signature is supposed to be, I looked
201    // at the internet for clues:
202    //
203    // https://stackoverflow.com/questions/46683342/securing-opensecurechannel-messages-and-x509identitytoken
204    // https://forum.prosysopc.com/forum/opc-ua/clarification-on-opensecurechannel-messages-and-x509identitytoken-specifications/
205    //
206    // These suggest that the signature is produced by appending the server nonce to the server certificate
207    // and signing with the user certificate's private key.
208    //
209    // This is the same as the standard handshake between client and server but using the identity cert. It would have been nice
210    // if the spec actually said this.
211
212    let signing_cert = super::x509::X509::from_byte_string(&token.certificate_data)?;
213    let result = super::verify_signature_data(
214        user_token_signature,
215        security_policy,
216        &signing_cert,
217        server_cert,
218        server_nonce,
219    );
220    if result.is_good() {
221        Ok(())
222    } else {
223        Err(result)
224    }
