33#[derive(Copy, Clone, PartialEq, Debug)]
34pub enum ConnectionState {
35    /// No connect has been made yet
36    NotStarted,
37    /// Connecting
38    Connecting,
39    /// Connection success
40    Connected,
41    // Waiting for ACK from the server
42    WaitingForAck,
43    // Connection is running
44    Processing,
45    // Connection is finished, possibly after an error
46    Finished(StatusCode),
47}
48
49#[derive(Clone)]
50/// A manager for the connection status with some helpers for common actions.
51pub(crate) struct ConnectionStateMgr {
52    state: Arc<RwLock<ConnectionState>>,
53}
54
55impl ConnectionStateMgr {
56    pub fn new() -> Self {
57        Self {
58            state: Arc::new(RwLock::new(ConnectionState::NotStarted)),
59        }
60    }
61
62    pub fn state(&self) -> ConnectionState {
63        let connection_state = trace_read_lock!(self.state);
64        *connection_state
65    }
66
67    pub fn set_state(&self, state: ConnectionState) {
68        trace!("setting connection state to {:?}", state);
69        let mut connection_state = trace_write_lock!(self.state);
70        *connection_state = state;
71    }
72
73    pub fn set_finished(&self, finished_code: StatusCode) {
74        self.set_state(ConnectionState::Finished(finished_code));
75    }
76
77    pub fn is_connected(&self) -> bool {
78        !matches!(
79            self.state(),
80            ConnectionState::NotStarted
81                | ConnectionState::Connecting
82                | ConnectionState::Finished(_)
83        )
84    }
85
86    pub fn is_finished(&self) -> bool {
87        matches!(self.state(), ConnectionState::Finished(_))
88    }
89}

95/// Session's state indicates connection status, negotiated times and sizes,
96/// and security tokens.
97pub(crate) struct SessionState {
98    /// A unique identifier for the session, this is NOT the session id assigned after a session is created
99    id: u32,
100    /// Time offset between the client and the server.
101    client_offset: Duration,
102    /// Ignore clock skew between the client and the server.
103    ignore_clock_skew: bool,
104    /// Secure channel information
105    secure_channel: Arc<RwLock<SecureChannel>>,
106    /// Connection state - what the session's connection is currently doing
107    connection_state: ConnectionStateMgr,
108    /// The request timeout is how long the session will wait from sending a request expecting a response
109    /// if no response is received the client will terminate.
110    request_timeout: u32,
111    /// Size of the send buffer
112    send_buffer_size: usize,
113    /// Size of the
114    receive_buffer_size: usize,
115    /// Maximum message size
116    max_message_size: usize,
117    /// Maximum chunk size
118    max_chunk_count: usize,
119    /// The session's id assigned after a connection and used for diagnostic info
120    session_id: NodeId,
121    /// The session authentication token, used for session activation
122    authentication_token: NodeId,
123    /// The next handle to assign to a request
124    request_handle: Handle,
125    /// Next monitored item client side handle
126    monitored_item_handle: Handle,
127    /// Subscription acknowledgements pending for send
128    subscription_acknowledgements: Vec<SubscriptionAcknowledgement>,
129    /// Subscription state
130    subscription_state: Arc<RwLock<SubscriptionState>>,
131    /// Connection closed callback
132    session_closed_callback: Option<Box<dyn OnSessionClosed + Send + Sync + 'static>>,
133    /// Connection status callback
134    connection_status_callback: Option<Box<dyn OnConnectionStatusChange + Send + Sync + 'static>>,
135    /// Message queue.
136    pub(crate) message_queue: Arc<RwLock<MessageQueue>>,
137}

140    fn on_session_closed(&mut self, status_code: StatusCode) {
141        debug!("Session was closed with status = {}", status_code);
142        if let Some(ref mut session_closed_callback) = self.session_closed_callback {
143            session_closed_callback.on_session_closed(status_code);
144        }
145    }

265    /// Construct a request header for the session. All requests after create session are expected
266    /// to supply an authentication token.
267    pub fn make_request_header(&mut self) -> RequestHeader {
268        RequestHeader {
269            authentication_token: self.authentication_token.clone(),
270            timestamp: DateTime::now_with_offset(self.client_offset),
271            request_handle: self.request_handle.next(),
272            return_diagnostics: DiagnosticBits::empty(),
273            timeout_hint: self.request_timeout,
274            ..Default::default()
275        }

278    /// Sends a publish request containing acknowledgements for previous notifications.
279    pub fn async_publish(&mut self) -> Result<u32, StatusCode> {
280        let subscription_acknowledgements = if self.subscription_acknowledgements.is_empty() {
281            None
282        } else {
283            let subscription_acknowledgements: Vec<SubscriptionAcknowledgement> =
284                self.subscription_acknowledgements.drain(..).collect();
285            // Debug sequence nrs
286            if log_enabled!(log::Level::Debug) {
287                let sequence_nrs: Vec<u32> = subscription_acknowledgements
288                    .iter()
289                    .map(|ack| ack.sequence_number)
290                    .collect();
291                debug!(
292                    "async_publish is acknowledging subscription acknowledgements with sequence nrs {:?}",
293                    sequence_nrs
294                );
295            }
296            Some(subscription_acknowledgements)
297        };
298        let request = PublishRequest {
299            request_header: self.make_request_header(),
300            subscription_acknowledgements,
301        };
302        let request_handle = self.async_send_request(request, None)?;
303
304        {
305            let mut subscription_state = trace_write_lock!(self.subscription_state);
306            subscription_state.set_last_publish_request(Instant::now());
307        }
308
309        debug!("async_publish, request sent with handle {}", request_handle);
310        Ok(request_handle)
311    }

313    /// Synchronously sends a request. The return value is the response to the request
314    pub(crate) fn send_request<T>(&mut self, request: T) -> Result<SupportedMessage, StatusCode>
315    where
316        T: Into<SupportedMessage>,
317    {
318        // A channel is created to receive the response
319        let (sender, receiver) = mpsc::sync_channel(1);
320        // Send the request
321        let request_handle = self.async_send_request(request, Some(sender))?;
322        // Wait for the response
323        let request_timeout = self.request_timeout();
324        self.wait_for_sync_response(request_handle, request_timeout, receiver)
325    }

327    pub(crate) fn reset(&mut self) {
328        // Clear tokens, ids etc.
329        self.session_id = NodeId::null();
330        self.authentication_token = NodeId::null();
331        self.request_handle.reset();
332        self.monitored_item_handle.reset();
333
334        // Clear the message queue
335        {
336            let mut message_queue = trace_write_lock!(self.message_queue);
337            message_queue.clear();
338        };
339    }

341    /// Asynchronously sends a request. The return value is the request handle of the request
342    pub(crate) fn async_send_request<T>(
343        &mut self,
344        request: T,
345        sender: Option<SyncSender<SupportedMessage>>,
346    ) -> Result<u32, StatusCode>
347    where
348        T: Into<SupportedMessage>,
349    {
350        let request = request.into();
351        match request {
352            SupportedMessage::OpenSecureChannelRequest(_)
353            | SupportedMessage::CloseSecureChannelRequest(_) => {}
354            _ => {
355                // Make sure secure channel token hasn't expired
356                let _ = self.ensure_secure_channel_token();
357            }
358        }
359
360        // TODO should error here if not connected
361
362        // Enqueue the request
363        let request_handle = request.request_handle();
364        self.add_request(request, sender);
365
366        Ok(request_handle)
367    }

374    /// Wait for a response with a matching request handle. If request handle is 0 then no match
375    /// is performed and in fact the function is expected to receive no messages except asynchronous
376    /// and housekeeping events from the server. A 0 handle will cause the wait to process at most
377    /// one async message before returning.
378    fn wait_for_sync_response(
379        &mut self,
380        request_handle: u32,
381        request_timeout: u32,
382        receiver: Receiver<SupportedMessage>,
383    ) -> Result<SupportedMessage, StatusCode> {
384        if request_handle == 0 {
385            panic!("Request handle must be non zero");
386        }
387        // Receive messages until the one expected comes back. Publish responses will be consumed
388        // silently.
389        let request_timeout = std::time::Duration::from_millis(request_timeout as u64);
390        receiver.recv_timeout(request_timeout).map_err(|_| {
391            info!("Timeout waiting for response from server");
392            self.request_has_timed_out(request_handle);
393            StatusCode::BadTimeout
394        })
395    }

397    fn request_has_timed_out(&self, request_handle: u32) {
398        let mut message_queue = trace_write_lock!(self.message_queue);
399        message_queue.request_has_timed_out(request_handle)
400    }

411    /// Checks if secure channel token needs to be renewed and renews it
412    fn ensure_secure_channel_token(&mut self) -> Result<(), StatusCode> {
413        let should_renew_security_token = {
414            let secure_channel = trace_read_lock!(self.secure_channel);
415            secure_channel.should_renew_security_token()
416        };
417        if should_renew_security_token {
418            self.issue_or_renew_secure_channel(SecurityTokenRequestType::Renew)
419        } else {
420            Ok(())
421        }
422    }

424    pub(crate) fn issue_or_renew_secure_channel(
425        &mut self,
426        request_type: SecurityTokenRequestType,
427    ) -> Result<(), StatusCode> {
428        trace!("issue_or_renew_secure_channel({:?})", request_type);
429
430        const REQUESTED_LIFETIME: u32 = 60000; // TODO
431
432        let (security_mode, security_policy, client_nonce) = {
433            let mut secure_channel = trace_write_lock!(self.secure_channel);
434            let client_nonce = secure_channel.security_policy().random_nonce();
435            secure_channel.set_local_nonce(client_nonce.as_ref());
436            (
437                secure_channel.security_mode(),
438                secure_channel.security_policy(),
439                client_nonce,
440            )
441        };
442
443        info!("Making secure channel request");
444        info!("security_mode = {:?}", security_mode);
445        info!("security_policy = {:?}", security_policy);
446
447        let requested_lifetime = REQUESTED_LIFETIME;
448        let request = OpenSecureChannelRequest {
449            request_header: self.make_request_header(),
450            client_protocol_version: 0,
451            request_type,
452            security_mode,
453            client_nonce,
454            requested_lifetime,
455        };
456        let response = self.send_request(request)?;
457        if let SupportedMessage::OpenSecureChannelResponse(response) = response {
458            // Extract the security token from the response.
459            let mut security_token = response.security_token.clone();
460
461            // When ignoring clock skew, we calculate the time offset between the client and the
462            // server and use that offset to compensate for the difference in time when setting
463            // the timestamps in the request headers and when decoding timestamps in messages
464            // received from the server.
465            if self.ignore_clock_skew && !response.response_header.timestamp.is_null() {
466                let offset = response.response_header.timestamp - DateTime::now();
467                // Make sure to apply the offset to the security token in the current response.
468                security_token.created_at = security_token.created_at - offset;
469                // Update the client offset by adding the new offset. When the secure channel is
470                // renewed its already using the client offset calculated when issuing the secure
471                // channel and only needs to be updated to accommodate any additional clock skew.
472                self.set_client_offset(offset);
473            }
474
475            debug!("Setting transport's security token");
476            {
477                let mut secure_channel = trace_write_lock!(self.secure_channel);
478                secure_channel.set_client_offset(self.client_offset);
479                secure_channel.set_security_token(security_token);
480
481                if security_policy != SecurityPolicy::None
482                    && (security_mode == MessageSecurityMode::Sign
483                        || security_mode == MessageSecurityMode::SignAndEncrypt)
484                {
485                    secure_channel.set_remote_nonce_from_byte_string(&response.server_nonce)?;
486                    secure_channel.derive_keys();
487                }
488            }
489            Ok(())
490        } else {
491            Err(process_unexpected_response(response))
492        }
493    }

495    // Process any async messages we expect to receive
496    pub(crate) fn handle_publish_responses(&mut self) -> bool {
497        let responses = {
498            let mut message_queue = trace_write_lock!(self.message_queue);
499            message_queue.async_responses()
500        };
501        if responses.is_empty() {
502            false
503        } else {
504            session_debug!(self, "Processing {} async messages", responses.len());
505            for response in responses {
506                self.handle_async_response(response);
507            }
508            true
509        }
510    }
511
512    /// This is the handler for asynchronous responses which are currently assumed to be publish
513    /// responses. It maintains the acknowledgements to be sent and sends the data change
514    /// notifications to the client for processing.
515    fn handle_async_response(&mut self, response: SupportedMessage) {
516        session_debug!(self, "handle_async_response");
517        match response {
518            SupportedMessage::PublishResponse(response) => {
519                session_debug!(self, "PublishResponse");
520
521                // Update subscriptions based on response
522                // Queue acknowledgements for next request
523                let notification_message = response.notification_message.clone();
524                let subscription_id = response.subscription_id;
525
526                // Queue an acknowledgement for this request (if it has data)
527                if let Some(ref notification_data) = notification_message.notification_data {
528                    if !notification_data.is_empty() {
529                        self.add_subscription_acknowledgement(SubscriptionAcknowledgement {
530                            subscription_id,
531                            sequence_number: notification_message.sequence_number,
532                        });
533                    }
534                }
535
536                let decoding_options = {
537                    let secure_channel = trace_read_lock!(self.secure_channel);
538                    secure_channel.decoding_options()
539                };
540
541                // Process data change notifications
542                if let Some((data_change_notifications, events)) =
543                    notification_message.notifications(&decoding_options)
544                {
545                    session_debug!(
546                        self,
547                        "Received notifications, data changes = {}, events = {}",
548                        data_change_notifications.len(),
549                        events.len()
550                    );
551                    if !data_change_notifications.is_empty() {
552                        let mut subscription_state = trace_write_lock!(self.subscription_state);
553                        subscription_state
554                            .on_data_change(subscription_id, &data_change_notifications);
555                    }
556                    if !events.is_empty() {
557                        let mut subscription_state = trace_write_lock!(self.subscription_state);
558                        subscription_state.on_event(subscription_id, &events);
559                    }
560                }
561
562                // Send another publish request
563                let _ = self.async_publish();
564            }
565            SupportedMessage::ServiceFault(response) => {
566                let service_result = response.response_header.service_result;
567                session_debug!(
568                    self,
569                    "Service fault received with {} error code",
570                    service_result
571                );
572                session_trace!(self, "ServiceFault {:?}", response);
573
574                match service_result {
575                    StatusCode::BadTimeout => {
576                        debug!("Publish request timed out so sending another");
577                        let _ = self.async_publish();
578                    }
579                    StatusCode::BadTooManyPublishRequests => {
580                        // Turn off publish requests until server says otherwise
581                        debug!("Server tells us too many publish requests so waiting for a response before resuming");
582                    }
583                    StatusCode::BadSessionClosed
584                    | StatusCode::BadSessionIdInvalid
585                    | StatusCode::BadNoSubscription
586                    | StatusCode::BadSubscriptionIdInvalid => {
587                        self.on_session_closed(service_result)
588                    }
589                    _ => (),
590                }
591            }
592            _ => {
593                info!("{} unhandled response", self.session_id());
594            }
595        }
596    }
