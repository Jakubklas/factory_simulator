5//! Provides subscription and monitored item tracking.
6//!
7//! The structs and functions in this file allow the client to maintain a shadow copy of the
8//! subscription and monitored item state on the server. If the server goes down and the session
9//! needs to be recreated, the client API can reconstruct the subscriptions and monitored item from
10//! its shadow version.
11//!
12//! None of this is for public consumption. The client is expected to recreate state automatically
13//! on a reconnect if necessary.
14

45#[derive(Debug)]
46pub struct MonitoredItem {
47    /// This is the monitored item's id within the subscription
48    id: u32,
49    /// Monitored item's handle. Used internally - not modifiable
50    client_handle: u32,
51    // The thing that is actually being monitored - the node id, attribute, index, encoding.
52    item_to_monitor: ReadValueId,
53    /// Queue size
54    queue_size: usize,
55    /// Discard oldest
56    discard_oldest: bool,
57    /// Monitoring mode
58    monitoring_mode: MonitoringMode,
59    /// Sampling interval
60    sampling_interval: f64,
61    /// Last value of the item
62    last_value: DataValue,
63    /// A list of all values received in the last data change notification. This list is cleared immediately
64    /// after the data change notification.
65    values: Vec<DataValue>,
66    /// Triggered items
67    triggered_items: BTreeSet<u32>,
68}
69
70impl MonitoredItem {
71    pub fn new(client_handle: u32) -> MonitoredItem {
72        MonitoredItem {
73            id: 0,
74            queue_size: 1,
75            sampling_interval: 0.0,
76            item_to_monitor: ReadValueId {
77                node_id: NodeId::null(),
78                attribute_id: 0,
79                index_range: UAString::null(),
80                data_encoding: QualifiedName::null(),
81            },
82            monitoring_mode: MonitoringMode::Reporting,
83            discard_oldest: false,
84            last_value: DataValue::null(),
85            values: Vec::with_capacity(1),
86            client_handle,
87            triggered_items: BTreeSet::new(),
88        }
89    }
90
91    pub fn id(&self) -> u32 {
92        self.id
93    }

95    pub fn client_handle(&self) -> u32 {

97    }

103    pub fn sampling_interval(&self) -> f64 {

106
107    pub fn queue_size(&self) -> usize {
108        self.queue_size
109    }
110
111    pub fn last_value(&self) -> &DataValue {
112        &self.last_value
113    }

115    pub fn values(&self) -> &Vec<DataValue> {
116        &self.values
117    }
118
119    pub fn clear_values(&mut self) {
120        self.values.clear();
121    }

123    pub fn append_new_value(&mut self, value: DataValue) {
124        if self.values.len() == self.queue_size {
125            let _ = self.values.pop();
126            self.values.push(value);
127        }
128    }
129
130    pub fn monitoring_mode(&self) -> MonitoringMode {
131        self.monitoring_mode
132    }
133
134    pub fn discard_oldest(&self) -> bool {
135        self.discard_oldest
136    }
137
138    pub(crate) fn set_id(&mut self, value: u32) {
139        self.id = value;
140    }
141
142    pub(crate) fn set_item_to_monitor(&mut self, item_to_monitor: ReadValueId) {
143        self.item_to_monitor = item_to_monitor;
144    }
145
146    pub(crate) fn set_sampling_interval(&mut self, value: f64) {
147        self.sampling_interval = value;
148    }
149
150    pub(crate) fn set_queue_size(&mut self, value: usize) {
151        self.queue_size = value;
152        if self.queue_size > self.values.capacity() {
153            self.values
154                .reserve(self.queue_size - self.values.capacity());
155        }
156    }
157
158    pub(crate) fn set_monitoring_mode(&mut self, monitoring_mode: MonitoringMode) {
159        self.monitoring_mode = monitoring_mode;
160    }
161
162    pub(crate) fn set_discard_oldest(&mut self, discard_oldest: bool) {
163        self.discard_oldest = discard_oldest;
164    }
165
166    pub(crate) fn set_triggering(&mut self, links_to_add: &[u32], links_to_remove: &[u32]) {
167        links_to_remove.iter().for_each(|i| {
168            self.triggered_items.remove(i);
169        });
170        links_to_add.iter().for_each(|i| {
171            self.triggered_items.insert(*i);
172        });
173    }
174
175    pub(crate) fn triggered_items(&self) -> &BTreeSet<u32> {
176        &self.triggered_items
177    }
178}

180pub struct Subscription {
181    /// Subscription id, supplied by server
182    subscription_id: u32,
183    /// Publishing interval in seconds
184    publishing_interval: f64,
185    /// Lifetime count, revised by server
186    lifetime_count: u32,
187    /// Max keep alive count, revised by server
188    max_keep_alive_count: u32,
189    /// Max notifications per publish, revised by server
190    max_notifications_per_publish: u32,
191    /// Publishing enabled
192    publishing_enabled: bool,
193    /// Priority
194    priority: u8,
195    /// The change callback will be what is called if any monitored item changes within a cycle.
196    /// The monitored item is referenced by its id
197    notification_callback: Arc<Mutex<dyn OnSubscriptionNotification + Send + Sync>>,
198    /// A map of monitored items associated with the subscription (key = monitored_item_id)
199    monitored_items: HashMap<u32, MonitoredItem>,
200    /// A map of client handle to monitored item id
201    client_handles: HashMap<u32, u32>,
202}

292    pub(crate) fn insert_monitored_items(&mut self, items_to_create: &[CreateMonitoredItem]) {
293        items_to_create.iter().for_each(|i| {
294            let mut monitored_item = MonitoredItem::new(i.client_handle);
295            monitored_item.set_id(i.id);
296            monitored_item.set_monitoring_mode(i.monitoring_mode);
297            monitored_item.set_discard_oldest(i.discard_oldest);
298            monitored_item.set_sampling_interval(i.sampling_interval);
299            monitored_item.set_queue_size(i.queue_size as usize);
300            monitored_item.set_item_to_monitor(i.item_to_monitor.clone());
301
302            let client_handle = monitored_item.client_handle();
303            let monitored_item_id = monitored_item.id();
304            self.monitored_items
305                .insert(monitored_item_id, monitored_item);
306            self.client_handles.insert(client_handle, monitored_item_id);
307        });
308    }
309
310    pub(crate) fn modify_monitored_items(&mut self, items_to_modify: &[ModifyMonitoredItem]) {
311        items_to_modify.iter().for_each(|i| {
312            if let Some(ref mut monitored_item) = self.monitored_items.get_mut(&i.id) {
313                monitored_item.set_sampling_interval(i.sampling_interval);
314                monitored_item.set_queue_size(i.queue_size as usize);
315            }
316        });
317    }
318
319    pub(crate) fn delete_monitored_items(&mut self, items_to_delete: &[u32]) {
320        items_to_delete.iter().for_each(|id| {
321            // Remove the monitored item and the client handle / id entry
322            if let Some(monitored_item) = self.monitored_items.remove(id) {
323                let _ = self.client_handles.remove(&monitored_item.client_handle());
324            }
325        })
326    }
327
328    pub(crate) fn set_triggering(
329        &mut self,
330        triggering_item_id: u32,
331        links_to_add: &[u32],
332        links_to_remove: &[u32],
333    ) {
334        if let Some(ref mut monitored_item) = self.monitored_items.get_mut(&triggering_item_id) {
335            monitored_item.set_triggering(links_to_add, links_to_remove);
336        }
337    }
338
339    fn monitored_item_id_from_handle(&self, client_handle: u32) -> Option<u32> {
340        self.client_handles.get(&client_handle).copied()
341    }
342
343    pub(crate) fn on_event(&mut self, events: &[EventNotificationList]) {
344        let mut cb = trace_lock!(self.notification_callback);
345        events.iter().for_each(|event| {
346            cb.on_event(event);
347        });
348    }
349
350    pub(crate) fn on_data_change(&mut self, data_change_notifications: &[DataChangeNotification]) {
351        let mut monitored_item_ids = HashSet::new();
352        data_change_notifications.iter().for_each(|n| {
353            if let Some(ref monitored_items) = n.monitored_items {
354                monitored_item_ids.clear();
355                for i in monitored_items {
356                    let monitored_item_id = {
357                        let monitored_item_id = self.monitored_item_id_from_handle(i.client_handle);
358                        if monitored_item_id.is_none() {
359                            continue;
360                        }
361                        *monitored_item_id.as_ref().unwrap()
362                    };
363                    let monitored_item = self.monitored_items.get_mut(&monitored_item_id).unwrap();
364                    monitored_item.last_value = i.value.clone();
365                    monitored_item.values.push(i.value.clone());
366                    monitored_item_ids.insert(monitored_item_id);
367                }
368                if !monitored_item_ids.is_empty() {
369                    let data_change_items: Vec<&MonitoredItem> = monitored_item_ids
370                        .iter()
371                        .map(|id| self.monitored_items.get(id).unwrap())
372                        .collect();
373
374                    {
375                        // Call the call back with the changes we collected
376                        let mut cb = trace_lock!(self.notification_callback);
377                        cb.on_data_change(&data_change_items);
378                    }
379
380                    // Clear the values
381                    monitored_item_ids.iter().for_each(|id| {
382                        let m = self.monitored_items.get_mut(id).unwrap();
383                        m.clear_values();
384                    });
385                }
386            }
387        });
388    }
