5use std::path::PathBuf;
6
7use crate::client::{client::Client, config::*};
8use crate::core::config::Config;

10/// The `ClientBuilder` is a builder for producing a [`Client`]. It is an alternative to constructing
11/// a [`ClientConfig`] from file or from scratch.
12///
13/// # Example
14///
15/// ```no_run
16/// use opcua::client::prelude::*;
17///
18/// fn main() {
19///     let builder = ClientBuilder::new()
20///         .application_name("OPC UA Sample Client")
21///         .application_uri("urn:SampleClient")
22///         .pki_dir("./pki")
23///         .endpoints(vec![
24///             ("sample_endpoint", ClientEndpoint {
25///                 url: String::from("opc.tcp://127.0.0.1:4855/"),
26///                 security_policy: String::from(SecurityPolicy::None.to_str()),
27///                 security_mode: String::from(MessageSecurityMode::None),
28///                 user_token_id: ANONYMOUS_USER_TOKEN_ID.to_string(),
29///             }),
30///         ])
31///         .default_endpoint("sample_endpoint")
32///         .create_sample_keypair(true)
33///         .trust_server_certs(true)
34///         .user_token("sample_user", ClientUserToken::user_pass("sample1", "sample1pwd"));
35///     let client = builder.client().unwrap();
36/// }
37/// ```
38///
39/// [`Client`]: ../client/struct.Client.html
40/// [`ClientConfig`]: ../config/struct.ClientConfig.html
41///
42pub struct ClientBuilder {
43    config: ClientConfig,
44}
45
46impl Default for ClientBuilder {
47    fn default() -> Self {
48        ClientBuilder {
49            config: ClientConfig::default(),
50        }
51    }
52}
53
54impl ClientBuilder {
55    /// Creates a `ClientBuilder`
56    pub fn new() -> ClientBuilder {
57        ClientBuilder::default()
58    }
59
60    /// Creates a `ClientBuilder` using a configuration file as the initial state.
61    pub fn from_config<T>(path: T) -> Result<ClientBuilder, ()>
62    where
63        T: Into<PathBuf>,
64    {
65        Ok(ClientBuilder {
66            config: ClientConfig::load(&path.into())?,
67        })
68    }
69
70    /// Yields a [`Client`] from the values set by the builder. If the builder is not in a valid state
71    /// it will return `None`.
72    ///
73    /// [`Client`]: ../client/struct.Client.html
74    pub fn client(self) -> Option<Client> {
75        if self.is_valid() {
76            Some(Client::new(self.config))
77        } else {
78            None
79        }
80    }
81
82    /// Yields a [`ClientConfig`] from the values set by the builder.
83    ///
84    /// [`ClientConfig`]: ../config/struct.ClientConfig.html
85    pub fn config(self) -> ClientConfig {
86        self.config
87    }
88
89    /// Tests if the builder is in a valid state to be able to yield a `Client`.
90    pub fn is_valid(&self) -> bool {
91        self.config.is_valid()
92    }
93
94    /// Sets the application name.
95    pub fn application_name<T>(mut self, application_name: T) -> Self
96    where
97        T: Into<String>,
98    {
99        self.config.application_name = application_name.into();
100        self
101    }
102
103    /// Sets the application uri
104    pub fn application_uri<T>(mut self, application_uri: T) -> Self
105    where
106        T: Into<String>,
107    {
108        self.config.application_uri = application_uri.into();
109        self
110    }
111
112    /// Sets the product uri.
113    pub fn product_uri<T>(mut self, product_uri: T) -> Self
114    where
115        T: Into<String>,
116    {
117        self.config.product_uri = product_uri.into();
118        self
119    }
120
121    /// Sets whether the client should generate its own key pair if there is none found in the pki
122    /// directory.
123    pub fn create_sample_keypair(mut self, create_sample_keypair: bool) -> Self {
124        self.config.create_sample_keypair = create_sample_keypair;
125        self
126    }
127
128    /// Sets a custom client certificate path. The path is required to be provided as a partial
129    /// path relative to the PKI directory. If set, this path will be used to read the client
130    /// certificate from disk. The certificate can be in either the .der or .pem format.
131    pub fn certificate_path<T>(mut self, certificate_path: T) -> Self
132    where
133        T: Into<PathBuf>,
134    {
135        self.config.certificate_path = Some(certificate_path.into());
136        self
137    }
138
139    /// Sets a custom private key path. The path is required to be provided as a partial path
140    /// relative to the PKI directory. If set, this path will be used to read the private key
141    /// from disk.
142    pub fn private_key_path<T>(mut self, private_key_path: T) -> Self
143    where
144        T: Into<PathBuf>,
145    {
146        self.config.private_key_path = Some(private_key_path.into());
147        self
148    }
149
150    /// Sets whether the client should automatically trust servers. If this is not set then
151    /// the client will reject the server upon first connect and the server's certificate
152    /// must be manually moved from pki's `/rejected` folder to the `/trusted` folder. If it is
153    /// set, then the server cert will automatically be stored in the `/trusted` folder.
154    pub fn trust_server_certs(mut self, trust_server_certs: bool) -> Self {
155        self.config.trust_server_certs = trust_server_certs;
156        self
157    }
158
159    /// Sets whether the client should verify server certificates. Regardless of this setting,
160    /// server certificates are always checked to see if they are trusted and have a valid key
161    /// length. In addition (if `verify_server_certs` is unset or is set to `true`) it will
162    /// verify the hostname, application uri and the not before / after values to ensure validity.
163    pub fn verify_server_certs(mut self, verify_server_certs: bool) -> Self {
164        self.config.verify_server_certs = verify_server_certs;
165        self
166    }
167
168    /// Sets the pki directory where client's own key pair is stored and where `/trusted` and
169    /// `/rejected` server certificates are stored.
170    pub fn pki_dir<T>(mut self, pki_dir: T) -> Self
171    where
172        T: Into<PathBuf>,
173    {
174        self.config.pki_dir = pki_dir.into();
175        self
176    }
177
178    /// Sets the preferred locales of the client. These are passed to the server during session
179    /// creation to ensure localized strings are in the preferred language.
180    pub fn preferred_locales(mut self, preferred_locales: Vec<String>) -> Self {
181        self.config.preferred_locales = preferred_locales;
182        self
183    }
184
185    /// Sets the id of the default endpoint to connect to.
186    pub fn default_endpoint<T>(mut self, endpoint_id: T) -> Self
187    where
188        T: Into<String>,
189    {
190        self.config.default_endpoint = endpoint_id.into();
191        self
192    }
193
194    /// Adds an endpoint to the list of endpoints the client knows of.
195    pub fn endpoint<T>(mut self, endpoint_id: T, endpoint: ClientEndpoint) -> Self
196    where
197        T: Into<String>,
198    {
199        self.config.endpoints.insert(endpoint_id.into(), endpoint);
200        self
201    }
202
203    /// Adds multiple endpoints to the list of endpoints the client knows of.
204    pub fn endpoints<T>(mut self, endpoints: Vec<(T, ClientEndpoint)>) -> Self
205    where
206        T: Into<String>,
207    {
208        for e in endpoints {
209            self.config.endpoints.insert(e.0.into(), e.1);
210        }
211        self
212    }
213
214    /// Adds a user token to the list supported by the client.
215    pub fn user_token<T>(mut self, user_token_id: T, user_token: ClientUserToken) -> Self
216    where
217        T: Into<String>,
218    {
219        let user_token_id = user_token_id.into();
220        if user_token_id == ANONYMOUS_USER_TOKEN_ID {
221            panic!("User token id {} is reserved", user_token_id);
222        }
223        self.config.user_tokens.insert(user_token_id, user_token);
224        self
225    }
226
227    /// Sets the session retry limit.
228    pub fn session_retry_limit(mut self, session_retry_limit: i32) -> Self {
229        if session_retry_limit < 0 && session_retry_limit != -1 {
230            panic!("Session retry limit must be -1, 0 or a positive number");
231        }
232        self.config.session_retry_limit = session_retry_limit;
233        self
234    }
235
236    /// Sets the session retry interval.
237    pub fn session_retry_interval(mut self, session_retry_interval: u32) -> Self {
238        self.config.session_retry_interval = session_retry_interval;
239        self
240    }
241
242    /// Sets the session timeout period.
243    pub fn session_timeout(mut self, session_timeout: u32) -> Self {
244        self.config.session_timeout = session_timeout;
245        self
246    }
247
248    /// Sets whether the client should ignore clock skew so the client can make a successful
249    /// connection to the server, even when the client and server clocks are out of sync.
250    pub fn ignore_clock_skew(mut self) -> Self {
251        self.config.performance.ignore_clock_skew = true;
252        self
253    }
254
255    /// Configures the client to use a single-threaded executor. This reduces the number of
256    /// threads used by the client.
257    pub fn single_threaded_executor(mut self) -> Self {
258        self.config.performance.single_threaded_executor = true;
259        self
260    }
261
262    /// Configures the client to use a multi-threaded executor.
263    pub fn multi_threaded_executor(mut self) -> Self {
264        self.config.performance.single_threaded_executor = false;
265        self
266    }
267
268    /// Session name - the default name to use for a new session
269    pub fn session_name<T>(mut self, session_name: T) -> Self
270    where
271        T: Into<String>,
272    {
273        self.config.session_name = session_name.into();
274        self
275    }
276
277    /// Set the maximum message size
278    pub fn max_message_size(mut self, max_message_size: usize) -> Self {
279        self.config.decoding_options.max_message_size = max_message_size;
280        self
281    }
282
283    /// Set the max chunk count
284    pub fn max_chunk_count(mut self, max_chunk_count: usize) -> Self {
285        self.config.decoding_options.max_chunk_count = max_chunk_count;
286        self
287    }
288}

290#[test]
291fn client_builder() {
292    use std::str::FromStr;
293
294    // The builder should produce a config that reflects the values that are explicitly set upon it.
295    let b = ClientBuilder::new()
296        .application_name("appname")
297        .application_uri("http://appname")
298        .product_uri("http://product")
299        .create_sample_keypair(true)
300        .certificate_path("certxyz")
301        .private_key_path("keyxyz")
302        .trust_server_certs(true)
303        .verify_server_certs(false)
304        .pki_dir("pkixyz")
305        .preferred_locales(vec!["a".to_string(), "b".to_string(), "c".to_string()])
306        .default_endpoint("http://default")
307        .session_retry_interval(1234)
308        .session_retry_limit(999)
309        .session_timeout(777)
310        .ignore_clock_skew()
311        .single_threaded_executor()
312        .session_name("SessionName")
313        // TODO user tokens, endpoints
314        ;
315
316    let c = b.config();
317
318    assert_eq!(c.application_name, "appname");
319    assert_eq!(c.application_uri, "http://appname");
320    assert_eq!(c.product_uri, "http://product");
321    assert_eq!(c.create_sample_keypair, true);
322    assert_eq!(c.certificate_path, Some(PathBuf::from("certxyz")));
323    assert_eq!(c.private_key_path, Some(PathBuf::from("keyxyz")));
324    assert_eq!(c.trust_server_certs, true);
325    assert_eq!(c.verify_server_certs, false);
326    assert_eq!(c.pki_dir, PathBuf::from_str("pkixyz").unwrap());
327    assert_eq!(
328        c.preferred_locales,
329        vec!["a".to_string(), "b".to_string(), "c".to_string()]
330    );
331    assert_eq!(c.default_endpoint, "http://default");
332    assert_eq!(c.session_retry_interval, 1234);
333    assert_eq!(c.session_retry_limit, 999);
334    assert_eq!(c.session_timeout, 777);
335    assert_eq!(c.performance.ignore_clock_skew, true);
336    assert_eq!(c.performance.single_threaded_executor, true);
337    assert_eq!(c.session_name, "SessionName");
338}
